REPLACE:movement_labor = { # Explicitly non-socialist labour movement
	category = movement_category_ideological

	ideology = ideology_proletarian_movement
	character_ideologies = {
		ideology_reformer
		ideology_radical
		ideology_social_democrat
		ideology_utopian
	}

	creation_trigger = {
		has_technology_researched = labor_movement
		# Intentionally not modernisation-gated, if someone manages to proletarianise without modernising this should have consequences
	}
	creation_weight = {
		value = 100
		if = {
			limit = {
				owner ?= { # If a movement of this type exists in a neighbouring country, make it more likely to spawn here as well
					any_neighbouring_state = {
						owner = {
							prestige > ROOT.owner.prestige
							any_political_movement = {
								is_political_movement_type = movement_labor
							}
						}
					}
				}
			}
			multiply = {
				value = 3
			}
		}
		if = {
			limit = {
				ig:ig_trade_unions = { ig_clout >= 0.05 }
			}
			multiply = {
				value = 1.1
			}
		}
		if = {
			limit = {
				owner ?= {
					OR = {
						has_law_or_variant = law_type:law_no_workers_rights
						has_law_or_variant = law_type:law_combination_acts
					}
				}
			}
			multiply = {
				value = 1.2
			}
		}
		if = {
			limit = {
				owner ?= {
					has_law_or_variant = law_type:law_guild_system
				}
			}
			multiply = {
				value = 0.5
			}
		}
		if = {
			limit = {
				owner ?= {
					any_political_movement = {
						OR = {
							is_political_movement_type = movement_anarchist
							is_political_movement_type = movement_socialist
							is_political_movement_type = movement_communist
						}
					}
				}
			}
			multiply = {
				value = 0.01
			}
		}
	}
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}

	character_support_trigger = {
		OR = {
			has_ideology = ideology:ideology_reformer
			has_ideology = ideology:ideology_radical
			has_ideology = ideology:ideology_social_democrat
			has_ideology = ideology:ideology_utopian
			is_interest_group_type = ig_trade_unions
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = {
				has_ideology = ideology:ideology_social_democrat
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_reformer
					has_ideology = ideology:ideology_radical
				}
			}
			multiply = {
				value = 2
			}
		}
		if = { # Labour movement is the natural habitat of non-socialist trade unions characters
			limit = {
				is_interest_group_type = ig_trade_unions
				has_socialist_ideology = no
			}
			multiply = {
				value = 2
			}
		}
		if = { # Communists will support the labour movement if they really, really have to
			limit = {
				owner ?= {
					any_political_movement = {
						OR = {
							is_political_movement_type = movement_anarchist
							is_political_movement_type = movement_socialist
							is_political_movement_type = movement_communist
						}
					}
				}
				OR = {
					has_ideology = ideology:ideology_anarchist
					has_ideology = ideology:ideology_vanguardist
					has_ideology = ideology:ideology_communist
				}
			}
			multiply = {
				value = 0.01
			}
		}
	}

	pop_support_trigger = {
		NOT = {
			is_pop_type = slaves
		}
	}

	pop_support_factors = {
		movement_support_high_urbanization
		movement_support_high_literacy
		movement_support_below_expected_sol
		movement_support_engineers
		movement_support_academics
		movement_support_laborers
		movement_support_machinists
		movement_support_clerks
	}
	pop_support_weight = {
		if = {
			limit = {
				exists = scope:political_movement
				scope:political_movement ?= {
						OR = {
							is_being_suppressed = yes
							is_being_bolstered = yes
						}
					}
				}
			if = {
				limit = {
					scope:political_movement = {
						is_being_suppressed = yes
					}
				}
				add = {
					value = suppression_base_value
					multiply = {
						value = -0.5
						add = owner.modifier:country_suppression_attraction_factor
					}
					desc = "concept_suppressed"
				}
			}
			else = {
				add = {
					value = bolstering_base_value
					multiply = {
						value = 0.5
						add = owner.modifier:country_bolster_attraction_factor
					}
					desc = "concept_bolstered"
				}
			}
		}
		add = {
			desc = "URBAN_STATE"
			value = 10
			multiply = state.state_urbanization_rate
		}

		if = {
			limit = {
				strata = upper
			}
			multiply = {
				value = 0.10
				desc = "UPPER_NO_ICON"
			}
		}
		else_if = {
			limit = {
				strata = middle
			}
			multiply = {
				value = 0.5
				desc = "MIDDLE_NO_ICON"
			}
			if = {
				limit = {
					is_pop_type = engineers
				}
				add = {
					value = 3
					desc = "POP_ENGINEERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = academics
				}
				add = {
					value = 6
					desc = "POP_ACADEMICS"
				}
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = laborers
				}
				add = {
					value = 25
					desc = "POP_LABORERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = machinists
				}
				add = {
					value = 30
					desc = "POP_MACHINISTS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = clerks
				}
				add = {
					value = 10
					desc = "POP_CLERKS"
				}
			}
			else = {
				add = {
					value = 6
					desc = "LOWER_NO_ICON"
				}
			}
		}

		if = {
			limit = {
				owner = {
					has_technology_researched = socialism
				}
			}
			if = {
				limit = {
					owner = {
						NOT = { has_law_or_variant = law_type:law_public_schools }
					}
				}
				multiply = 1.2
			}
			if = {
				limit = {
					owner = {
						NOT = { has_law_or_variant = law_type:law_public_health_insurance }
					}
				}
				multiply = 1.2
			}
			if = {
				limit = {
					owner = {
						has_law_or_variant = law_type:law_no_workers_rights
					}
				}
				multiply = 1.2
			}
		}

		if = {
			limit = {
				standard_of_living < state.average_expected_sol # It is intentional that a labourer being well-off doesn't decrease support for this.
			}
			add = { # Increases likelihood of support if impoverished.
				desc = "IS_BELOW_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				max = 10
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}

		if = { # If a more radical movement exists, it'll poach radicals from this one.
			limit = {
				owner ?= {
					any_political_movement = {
						OR = {
							is_political_movement_type = movement_socialist
							is_political_movement_type = movement_communist
							is_political_movement_type = movement_anarchist
						}
					}
				}
			}
			multiply = {
				desc = "POP_RADICAL_MOVEMENT_LOSSES"
				value = 1
				subtract = pop_radical_fraction
			}
		}
	}

	revolution = {
		possible = {
			political_movement_support >= 0.05
		}

		weight = {
			value = 100
		}

		state_weight = {
			value = state_urbanization_rate
		}
	}

	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}

		weight = {
			value = 1000
		}

		state_weight = {
			value = state_urbanization_rate
		}
	}

	additional_radicalism_factors = {
		owner = {
			if = {
				limit = { has_institution = institution_social_security }

				add = {
					value = 3
					subtract = "institution_investment_level(institution_social_security)"
					multiply = 0.05
					desc = "INSTITUTION_FUNDING_LEVEL_social_security"
				}
			}
			else = {
				add = {
					value = 0.1
					desc = "NO_INSTITUTION_social_security"
				}
			}

			if = {
				limit = { has_institution = institution_workplace_safety }

				add = {
					value = 3
					subtract = "institution_investment_level(institution_workplace_safety)"
					multiply = 0.05
					desc = "INSTITUTION_FUNDING_LEVEL_workplace_safety"
				}
			}
			else = {
				add = {
					value = 0.1
					desc = "NO_INSTITUTION_workplace_safety"
				}
			}
		}

		if = {
			limit = {
				owner = {
					any_neighbouring_state = {
						owner = {
							NOT = { THIS = ROOT.owner }
							any_political_movement = {
								is_political_movement_type = movement_labor
								is_insurrectionary = yes
							}
						}
					}
				}
			}
			add = {
				desc = "NEIGHBOURING_COUNTRY_WITH_REVOLUTION"
				value = 0.15
			}
		}
	}

	disband_trigger = {
		any_political_movement = {
			OR = {
				is_political_movement_type = movement_communist
				is_political_movement_type = movement_socialist
				is_political_movement_type = movement_anarchist
			}
			political_movement_support >= 0.1
		}
		exists = scope:political_movement
		scope:political_movement ?= {
			political_movement_support < 0.02
		}
	}
}

REPLACE:movement_cultural_majority = {
	category = movement_category_cultural_majority

	ideology = ideology_ethno_nationalist_movement
	character_ideologies = {
		ideology_ethno_nationalist
	}

	can_cause_obstinance = yes

	creation_trigger = {
		NOR = { # Will not spawn if you're an un-unified German or Italian state
			AND = {
				OR = {
					country_has_primary_culture = cu:north_german
					country_has_primary_culture = cu:south_german
				}
				NOT = {
					c:GER ?= this
				}
			}
			AND = {
				OR = {
					country_has_primary_culture = cu:north_italian
					country_has_primary_culture = cu:south_italian
				}
				NOR = {
					c:ITA ?= this
					c:AUS ?= this
					c:KUK ?= this
				}
			}
		}
	}

	creation_weight = {
		value = 100
	}
	on_created = {}

	character_support_trigger = {
		culture = {
			is_primary_culture_of = root.owner
		}
		OR = {
			has_ideology = ideology:ideology_fascist
			has_ideology = ideology:ideology_jingoist_leader
			has_ideology = ideology:ideology_ethno_nationalist
		}
		NOT = {
			has_variable = anti_nationalist_var
		}
	}
	character_support_weight = {
		value = 100
		if = {
			limit = {
				has_ideology = ideology:ideology_ethno_nationalist
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_fascist
					has_ideology = ideology:ideology_jingoist_leader
				}
			}
			multiply = {
				value = 2
			}
		}
		else_if = {
			limit = {
				OR = {
					has_progressive_ideology = yes
					has_socialist_ideology = yes
				}
			}
			multiply = 0.05
		}
	}

	pop_support_trigger = {
		culture = {
			is_primary_culture_of = root.owner
		}
	}

	pop_support_factors = {
		movement_support_national_awakening
		movement_support_cultural_fervor
		movement_support_shopkeepers
		movement_support_officers
		movement_support_farmers
		movement_support_clerks
		movement_support_soldiers
		movement_support_peasants
		movement_support_colonial_settlers
		movement_support_nationalism_tech
		movement_support_pan_nationalism_tech
	}

	pop_support_weight = {
		if = {
			limit = {
				exists = scope:political_movement
				scope:political_movement ?= {
					OR = {
						is_being_suppressed = yes
						is_being_bolstered = yes
					}
				}
			}
			if = {
				limit = {
					scope:political_movement = {
						is_being_suppressed = yes
					}
				}
				add = {
					value = suppression_base_value
					multiply = {
						value = -0.5
						add = owner.modifier:country_suppression_attraction_factor
					}
					desc = "concept_suppressed"
				}
			}
			else = {
				add = {
					value = bolstering_base_value
					multiply = {
						value = 0.5
						add = owner.modifier:country_bolster_attraction_factor
					}
					desc = "concept_bolstered"
				}
			}
		}

		if = {
			limit = {
				strata = upper
			}
			add = {
				value = 10
				desc = "UPPER_NO_ICON"
			}
		}
		else_if = {
			limit = {
				strata = middle
			}
			if = {
				limit = {
					is_pop_type = shopkeepers
				}
				add = {
					value = 25
					desc = "POP_SHOPKEEPERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = officers
				}
				add = {
					value = 20
					desc = "POP_OFFICERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = farmers
				}
				add = {
					value = 15
					desc = "POP_FARMERS"
				}
			}
			else = {
				add = {
					value = 12
					desc = "MIDDLE_NO_ICON"
				}
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = clerks
				}
				add = {
					value = 25
					desc = "POP_CLERKS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = soldiers
				}
				add = {
					value = 20
					desc = "POP_SOLDIERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = peasants
				}
				add = {
					value = 10
					desc = "POP_PEASANTS"
				}
			}
			else = {
				add = {
					value = 5
					desc = "LOWER_NO_ICON"
				}
			}
		}

		if = {
			limit = {
				state = {
					is_incorporated = no
				}
			}
			multiply = {
				value = 1.5
				desc = "COLONIAL_SETTLERS"
			}
		}

		if = {
			limit = {
				owner ?= {
					has_technology_researched = pan-nationalism
				}
			}
			multiply = {
				value = 1.5
				desc = "PAN_NATIONALISM_TECH"
			}
		}
		else_if = {
			limit = {
				owner ?= {
					has_technology_researched = nationalism
				}
			}
			multiply = {
				value = 1.25
				desc = "NATIONALISM_TECH"
			}
		}

		multiply = { # Range from x0.775 to 2.275
			desc = "POP_CULTURE_FERVOR"
			value = culture.culture_current_fervor
			subtract = 15
			multiply = 0.01
			add = 1
		}

		if = {
			limit = {
				culture ?= {
					culture_has_national_awakening = yes
				}
			}
			multiply = {
				value = 1.5
				desc = "POP_ACTIVE_NATIONAL_AWAKENING"
			}
		}

		if = { # If a more radical movement exists, it'll poach radicals from this one.
			limit = {
				owner ?= {
					any_political_movement = {
						OR = {
							is_political_movement_type = movement_reactionary
							is_political_movement_type = movement_fascist
						}
					}
				}
			}
			multiply = {
				desc = "POP_RADICAL_MOVEMENT_LOSSES"
				value = 1
				subtract = pop_radical_fraction
			}
		}
	}

	revolution = {
		possible = {
			political_movement_support >= 0.05
		}

		weight = {
			value = 100
		}

		state_weight = {
			value = 1

			if = {
				limit = {
					is_capital = no
					is_homeland_of_country_cultures = owner
					primary_cultures_percent_state >= 0.05
				}
				add = 10

				if = {
					limit = {
						primary_cultures_percent_state >= 0.75
					}
					add = 1000
				}
				else_if = {
					limit = {
						primary_cultures_percent_state >= 0.5
					}
					add = 500
				}
				else_if = {
					limit = {
						primary_cultures_percent_state >= 0.25
					}
					add = 100
				}
				else_if = {
					limit = {
						primary_cultures_percent_state >= 0.1
					}
					add = 50
				}
			}
		}
	}

	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}

		weight = {
			value = 1000
		}

		state_weight = {
			value = 0

			if = {
				limit = {
					is_capital = no
					is_homeland_of_country_cultures = owner
					primary_cultures_percent_state >= 0.05
				}
				add = { # These three values must add up to more than 1 - 5 x Obstinance, 1/2 Turmoil, percent of pops of primary cultures
					value = -1
					add = "this.primary_cultures_percent_state"
					add = {
						value = this.turmoil
						multiply = 0.5
					}
					add = {
						value = this.obstinance
						multiply = 5
					}
					multiply = 100
				}
			}
		}
	}

	additional_radicalism_factors = {
		if = {
			limit = {
				owner.ruler ?= {
					NOT = { culture = { is_primary_culture_of = root.owner } }
				}
			}

			add = {
				value = 0.1
				desc = "RULER_NOT_PRIMARY_CULTURE"
			}
		}

		add = {
			value = "owner.average_sol"
			subtract = "owner.average_sol_for_primary_cultures"
			multiply = 0.05
			max = 0.5
			min = 0
			desc = "STANDARD_OF_LIVING_FOR_PRIMARY_CULTURES"
		}

		if = {
			limit = {
				owner = {
					any_neighbouring_state = {
						owner = {
							NOT = { THIS = ROOT.owner }
							any_political_movement = {
								culture ?= ROOT.culture
								is_insurrectionary = yes
							}
						}
					}
				}
			}
			add = {
				desc = "NEIGHBOURING_COUNTRY_WITH_CULTURE_REVOLUTION"
				value = 0.15
			}
		}

		if = {
			limit = {
				owner = {
					has_law_or_variant = law_type:law_directorate
				}
			}
			multiply = 0.5
		}
	}

	disband_trigger = {
		any_political_movement = {
			is_political_movement_type = movement_fascist
			political_movement_support >= 0.1
		}
		exists = scope:political_movement
		scope:political_movement ?= {
			political_movement_support < 0.02
		}
	}
}

REPLACE:movement_reactionary = {
	category = movement_category_ideological

	ideology = ideology_reactionary_movement
	character_ideologies = {
		ideology_traditionalist
		ideology_protectionist
		ideology_authoritarian
		ideology_luddite
	}

	creation_trigger = {
		country_should_exclusively_focus_on_modernization = no
		has_law_or_variant = law_type:law_slavery_banned
		country_has_voting_franchise = yes
		NOT = { # Reactionary and Legitimist movements cannot co-exist
			any_political_movement = {
				OR = {
					is_political_movement_type = movement_legitimist
					is_political_movement_type = movement_carlist
					is_political_movement_type = movement_miguelist
				}
			}
		}
	}
	creation_weight = {
		value = 100
		if = {
			limit = {
				owner ?= { # If a movement of this type exists in a neighbouring country, make it more likely to spawn here as well
					any_neighbouring_state = {
						owner = {
							prestige > ROOT.owner.prestige
							any_political_movement = {
								is_political_movement_type = movement_reactionary
							}
						}
					}
				}
			}
			multiply = {
				value = 2
			}
		}
	}
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}

	disband_trigger = {
		OR = {
			AND = {

promote_monopolism_button = {
	name = "monopolism_button"
	desc = "monopolism_button_desc"

	visible = {
		has_variable = chose_initial_propertarian_philosophy
		NOT = { has_variable = boardroom_schism_freeze }
	}

	possible = {
		any_interest_group = {
			is_in_government = yes
			leader = {
				has_ideology = ideology:ideology_monopolist_leader
			}
		}
		custom_tooltip = {
			text = legitimate_government_tt
			legitimacy >= 25
		}
		custom_tooltip = {
			text = not_already_supporting_monopolism
			NOT = { has_variable = supporting_monopolism_var }
		}
		custom_tooltip = {
			text = not_recently_suspended_lobbying_tt
			OR = {
				has_variable = paused_lobbying_var
				NOT = { has_variable = pause_lobby_cooldown }
			}
		}
		custom_tooltip = {
			text = support_selection_on_cooldown_tt
			OR = {
				has_variable = paused_lobbying_var
				NOT = { has_variable = support_selection_cooldown }
			}
		}
	}

	selected = {
		has_variable = supporting_monopolism_var
		NOT = {
			has_variable = paused_lobbying_var
		}
	}

	effect = {
		set_variable = supporting_monopolism_var
		remove_variable = supporting_molinarism_var
		remove_variable = supporting_minarchism_var
		set_variable = {
			name = support_selection_cooldown
			value = 12
		}
		custom_tooltip = supporting_monopolism_tt
		custom_tooltip = supporting_monopolism_2_tt
		custom_tooltip = {
			text = monopolism_progress_increase_10_tt
			je:je_boardroom_schism = {
				add_progress = { value = 10 name = schism_monopolism_bar }
			}
		}
		custom_tooltip = support_selection_cooldown_tt
		if = {
			limit = {
				has_modifier = ir_suspended_lobbying_modifier
			}
			remove_modifier = ir_suspended_lobbying_modifier
		}
		if = {
			limit = { has_variable = paused_lobbying_var }
			remove_variable = paused_lobbying_var
			set_variable = {
				name = resume_lobby_cooldown
				value = 12
			}
		}
	}
}

suspend_lobbying_button = {
	name = "suspend_lobbying_button"
	desc = "suspend_lobbying_button_desc"

	visible = {
		has_variable = chose_initial_propertarian_philosophy
		NOT = { has_variable = boardroom_schism_freeze }
	}

	possible = {
		custom_tooltip = {
			text = not_already_suspended_lobbying_tt
			NOT = {
				has_variable = paused_lobbying_var
			}
		}
		custom_tooltip = {
			text = not_recently_resumed_lobbying_tt
			trigger_if = {
				NOT = {
					has_variable = resume_lobby_cooldown
				}
			}
		}
	}

	selected = {
		has_variable = paused_lobbying_var
	}

	effect = {
		set_variable = paused_lobbying_var
		set_variable = {
			name = pause_lobby_cooldown
			value = 12
		}
		custom_tooltip = suspended_lobbying_tt
		custom_tooltip = suspended_lobbying_1_tt
		custom_tooltip = suspended_lobbying_2_tt
		remove_variable = supporting_monopolism_var
		remove_variable = supporting_molinarism_var
		remove_variable = supporting_minarchism_var
		add_modifier = ir_suspended_lobbying_modifier
	}
}

sponsor_thinktanks_button = {
	name = "sponsor_thinktanks_button"
	desc = "sponsor_thinktanks_button_desc"

	visible = {
		has_variable = chose_initial_propertarian_philosophy
		NOT = { has_variable = boardroom_schism_freeze }
	}

	cooldown = {
		days = normal_modifier_time
	}

	possible = {
		has_law = law_type:law_private_schools
		NOT = {
			has_law = law_type:law_outlawed_dissent
		}
		custom_tooltip = {
			text = has_gold_reserves
			scaled_gold_reserves > 0
		}
		custom_tooltip = {
			text = not_sponsored_thinktanks_tt
			NOT = {
				has_variable = sponsored_thinktanks
			}
		}
		je:je_boardroom_schism = {
			custom_tooltip = {
				text = monopolism_progress_25
				scripted_bar_progress = { name = schism_monopolism_bar value >= 50 }
			}
			custom_tooltip = {
				text = minarchism_progress_25
				scripted_bar_progress = { name = schism_minarchism_bar value >= 50 }
			}
			custom_tooltip = {
				text = molinarism_progress_25
				scripted_bar_progress = { name = schism_molinarism_bar value >= 50 }
			}
		}
	}

	effect = {
		set_variable = sponsored_thinktanks
		add_modifier = {
			name = ir_sponsored_thinktanks_modifier
			years = 5
		}
		trigger_event = {
			id = boardroom_schism.3
			popup = yes
			months = 6
		}
		add_treasury = {
			value = gold_reserves_limit
			multiply = -0.25
		}
		custom_tooltip = {
			text = sponsored_thinktanks_progress_tt
			change_variable = {
				name = minarchist_effects_var
				add = 1
			}
			change_variable = {
				name = molinarist_effects_var
				add = 1
			}
			change_variable = {
				name = monopolist_effects_var
				add = 1
			}
		}
	}
}
convene_egm_button = {
	name = "convene_egm_button"
	desc = "convene_egm_button_desc"

	visible = {
		has_variable = chose_initial_propertarian_philosophy
		NOT = {
			has_variable = boardroom_schism_freeze
		}
	}

	possible = {
		trigger_if = {
			limit = {
				NOT = {
					has_variable = convened_egm
				}
			}
			gold_reserves > {
				value = gold_reserves_limit
				multiply = 0.75
			}
		}
		custom_tooltip = {
			text = not_convened_egm
			NOT = {
				has_variable = convened_egm
			}
		}
		trigger_if = {
			limit = {
				NOT = {
					has_variable = convened_egm
				}
			}
			je:je_boardroom_schism = {
				OR = {
					custom_tooltip = {
						text = monopolism_progress_400
						scripted_bar_progress = { name = schism_monopolism_bar value >= 400 }
					}
					custom_tooltip = {
						text = minarchism_progress_400
						scripted_bar_progress = { name = schism_minarchism_bar value >= 400 }
					}
					custom_tooltip = {
						text = molinarism_progress_400
						scripted_bar_progress = { name = schism_molinarism_bar value >= 400 }
					}
				}
			}
		}
	}

	effect = {
		trigger_event = {
			id = boardroom_schism.2
			popup = yes
			days = 7
		}
		set_variable = convened_egm
	}
	ai_chance = {
		value = 0
	}
}

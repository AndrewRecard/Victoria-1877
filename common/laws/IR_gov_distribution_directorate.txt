law_executive_absolutism = {
	group = lawgroup_distribution_of_power
	parent = law_autocracy

	icon = "gfx/interface/icons/law_icons/executive_absolutism.dds"

	progressiveness = -100

	is_visible = {
		has_law_or_variant = law_type:law_directorate
	}

	can_enact = {
		OR = {
		    has_law = law_type:law_interventionism
    		has_law = law_type:law_sovereign_monopoly
		}
		custom_tooltip = {
			text = completed_public_liquidation_tt
			has_variable = public_liquidation_completed
		}
	}

	on_activate = {
		scripted_effect_parties_disappearence = yes
		cleanup_voting_strength_modifiers = yes
		custom_tooltip = fixes_civil_service_tt
		if = {
			limit = {
				has_modifier = ir_guilded_charter_modifier
			}
			remove_modifier = ir_guilded_charter_modifier
		}
		hidden_effect = {
			trigger_event = {
				id = caciquismo.4
				popup = yes
				days = 7
			}
		}
	}

	modifier = {
		country_legitimacy_headofstate_add = 30
		country_legitimacy_govt_total_clout_add = 120
		country_legitimacy_ideological_incoherence_mult = 0.5
		country_capitalists_pol_str_mult = 0.5
		country_officers_pol_str_mult = 0.5
		country_authority_add = 250
		country_institution_corporate_unions_max_investment_add = 2
		country_allow_ruler_selection_bool = yes
	}

	institution = institution_civil_service
	institution_modifier = {
		state_tax_collection_mult = 0.01
	}

	can_impose = {
		OR = {
			AND = {
				is_in_same_power_bloc = scope:target_country
				AND = {
					modifier:country_can_impose_same_lawgroup_distribution_of_power_in_power_bloc_bool = yes
					has_law = scope:law
				}
			}
			can_impose_law_default = yes
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}

	ai_impose_chance = {
		value = 0

		if = {
			limit = {
				has_law = law_type:law_autocracy
				OR = {
					AND = {
						power_bloc ?= { has_identity = identity:identity_ideological_union }
						has_strategy = ai_strategy_conservative_agenda
					}
					has_strategy = ai_strategy_reactionary_agenda
				}
				scope:target_country = {
					NOR = { # Is their law different enough from what we want to impose?
						has_law_or_variant = law_type:law_oligarchy
						has_law_or_variant = law_type:law_landed_voting
					}
				}
			}

			add = base_impose_law_weight
		}

		if = {
			limit = {
				scope:target_country = {
					c:BIC ?= ROOT
					OR = {
						has_journal_entry = je_british_dictate_plantations
						has_journal_entry = je_british_dictate_military
						has_journal_entry = je_british_dictate_universities
					}
				}
			}
			multiply = 0
		}
	}
}

REPLACE:law_women_own_property = {
	group = lawgroup_rights_of_women
	icon = "gfx/interface/icons/law_icons/women_own_property.dds"

	progressiveness = -50

	on_enact = {
		recalculate_pop_ig_support = yes
	}

	modifier = {
		country_authority_add = 50
		state_working_adult_ratio_add = 0.03
		state_dependent_wage_mult = 0.1
		state_birth_rate_mult = 0.03
	}

	ai_will_do = {
		OR = {
			has_law_or_variant = law_type:law_no_womens_rights
			AND = {
				exists = ruler
				ruler = {
					has_ideology = ideology:ideology_traditionalist
				}
			}
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		if = {
			limit = { 
				has_law_or_variant = law_type:law_no_womens_rights
				OR = {
					ig:ig_intelligentsia = { is_in_government = yes }
					ig:ig_trade_unions = { is_in_government = yes }
					ig:ig_devout = { is_marginal = yes }
				}
			}
			add = 1000
		}
	}
}

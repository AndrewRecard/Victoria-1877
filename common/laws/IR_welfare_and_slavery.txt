REPLACE:law_old_age_pension = {
	group = lawgroup_welfare

	icon = "gfx/interface/icons/law_icons/old_age_pension.dds"

	progressiveness = 100

	unlocking_technologies = {
		human_rights
	}

	can_enact = {
		custom_tooltip = {
			text = public_institutions_liquidated_tt
			modifier:country_public_institutions_liquidated_bool = no
		}
	}

	modifier = {
		country_institution_cost_institution_social_security_mult = 0.25
		state_working_adult_ratio_add = -0.02
	}
	institution = institution_social_security
	institution_modifier = {
		state_welfare_payments_add = -0.05
		state_dependent_wage_mult = 0.15
		state_dependent_political_participation_add = 0.02
		state_food_security_add = 0.01
		building_minimum_wage_mult = 0.08
		state_political_strength_from_welfare_mult = 0.05
	}

	on_activate = {
		recalculate_pop_ig_support = yes
		if = {
			limit = {
				NOT = {
					has_variable = entrenched_entitlements_var
				}
			}
			set_variable = {
				name = entrenched_entitlement_years
				value = 0
			}
		}
		random_law = {
			limit = {
				type = law_type:law_old_age_pension
			}
			save_scope_as = relevant_welfare_law
		}
		inherit_entrenched_entitlements_effect = yes
	}

	on_deactivate = {
		recalculate_pop_ig_support = yes
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
}

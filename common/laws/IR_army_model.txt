law_private_defense_agencies = {
	group = lawgroup_army_model
	icon = "gfx/interface/icons/law_icons/private_defense_agencies.dds"

	progressiveness = 0

	unlocking_technologies = {
		military_statistics
	}

	disallowing_laws = {
		law_guild_system
		law_serfdom
		law_traditionalism
		law_cooperative_ownership
		law_council_republic
		law_strict_edo_system
		law_intermediate_edo_system
	}

	is_visible = {
		always = yes
	}

	can_enact = {
		NOR = {
			has_law_or_variant = law_type:law_council_republic
			is_involved_in_journal_entry = je_cre_undo_evolutions
		}
	}

	modifier = {
		country_influence_mult = -0.25
		country_max_companies_add = 1
		country_machinists_pol_str_mult = 0.25
		country_officers_pol_str_mult = 0.25
		state_building_barrack_max_level_add = 100
		country_military_wages_mult = 0.25
		building_group_bg_military_industry_throughput_add = 0.1
		unit_kill_rate_add = 0.1
		unit_experience_gain_mult = 1.0
		unit_devastation_mult = 0.5
		unit_morale_loss_mult = 0.1
		country_barracks_scaling_turmoil_bool = yes
		country_militarized_police_discount_bool = yes
	}

	can_impose = {
		OR = {
			AND = {
				is_in_same_power_bloc = scope:target_country
				AND = {
					modifier:country_can_impose_same_lawgroup_army_model_in_power_bloc_bool = yes
					has_law = scope:law
				}
			}
			can_impose_law_default = yes
		}
	}

	on_activate = {
		add_journal_entry = {
			type = je_defense_incorporation
		}
	}

	on_deactivate = {
		if = {
			limit = {
				has_modifier = ir_defense_agencies_drain_modifier
			}
			remove_modifier = ir_defense_agencies_drain_modifier
		}
	}

	on_enact = {
		if = {
			limit = {
				has_law_or_variant = law_type:law_peasant_levies
			}
			custom_tooltip = aristocrats_af_ig_shift_effect_negative_tt
		}
		if = {
			limit = {
				NOT = {
					has_law_or_variant = law_type:law_professional_army
				}
			}
			custom_tooltip = non_military_pops_af_ig_shift_effect_tt
			custom_tooltip = military_pops_af_ig_shift_effect_tt
		}
		custom_tooltip = machinists_af_ig_shift_effect_tt
		custom_tooltip = unlocks_propertarian_paramilitary_tt
		custom_tooltip = propertarian_armed_forces_attraction_tt
	}

	ai_will_do = {
		always = yes
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}

		if = {
			limit = {
				has_law_or_variant = law_type:law_directorate
			}
			add = 1000
		}

		if = {
			limit = {
				has_law_or_variant = law_type:law_laissez_faire
			}
			add = 20
		}

		if = {
			limit = {
				has_journal_entry = je_public_liquidation
				has_law_or_variant = law_type:law_militarized_police
			}
			add = 5000
		}
	}
}

REPLACE:law_peasant_levies = {
	group = lawgroup_army_model
	icon = "gfx/interface/icons/law_icons/peasant_levies.dds"

	progressiveness = -50

	is_visible = {
		NOT = {
			country_has_primary_culture = cu:japanese
		}
	}

	unlocking_laws = {
		law_traditionalism
		law_hereditary_bureaucrats
	}

	disallowing_laws = {
		law_council_republic
	}
	can_enact = {
		custom_tooltip = {
			text = public_institutions_liquidated_tt
			modifier:country_public_institutions_liquidated_bool = no
		}
	}

	modifier = {
		country_can_only_conscript_peasants_bool = yes
		unit_morale_loss_mult = 0.1
		unit_experience_gain_mult = -0.25
		state_conscription_rate_add = 0.04
		country_aristocrats_pol_str_mult = 0.25
		country_military_goods_cost_mult = -0.1
		state_building_barrack_max_level_add = 25
		state_building_conscription_center_max_level_add = 25
	}

	can_impose = {
		OR = {
			AND = {
				is_in_same_power_bloc = scope:target_country
				AND = {
					modifier:country_can_impose_same_lawgroup_army_model_in_power_bloc_bool = yes
					has_law = scope:law
				}
			}
			can_impose_law_default = yes
		}
	}

	on_enact = {
		custom_tooltip = QUALIFICATIONS_OFFICERS_PEASANT_LEVIES_DESC
		if = {
			limit = {
				NOT = {
					has_law_or_variant = law_type:law_peasant_levies
				}
			}
			custom_tooltip = aristocrats_af_ig_shift_effect_tt
		}
		if = {
			limit = {
				OR = {
					has_law_or_variant = law_type:law_professional_army
					has_law_or_variant = law_type:law_private_defense_agencies
				}
			}
			custom_tooltip = military_pops_af_ig_shift_effect_negative_tt
			custom_tooltip = non_military_pops_af_ig_shift_effect_negative_tt
		}
		if = {
			limit = {
				has_law_or_variant = law_type:law_private_defense_agencies
			}
			custom_tooltip = machinists_af_ig_shift_effect_negative_tt
		}
	}

	on_activate = {
		if = {
			limit = {
				ig:ig_armed_forces ?= {
					has_ideology = ideology:ideology_martial_corporatist
				}
			}
			ir_defense_contractors_reversal_effect = yes
			add_modifier = {
				name = ir_defense_agencies_reform_modifier
				years = 5
				is_decaying = yes
			}
		}
		if = {
			limit = {
				modifier:country_public_institutions_liquidated_bool = yes
				has_law_or_variant = law_type:law_militarized_police
			}
			activate_law = law_type:law_local_police
		}
	}

	# AI will never return to peasant levies
	ai_will_do = {
		always = no
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
}

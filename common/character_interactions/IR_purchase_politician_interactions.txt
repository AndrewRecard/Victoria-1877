sway_to_minarchism = {
	icon = "gfx/interface/icons/ideology_icons/ideology_leader/unused/ideology_leader_isolationist.dds"
	potential = {
		owner ?= scope:actor
		owner ?= {
			has_journal_entry = je_boardroom_schism
			has_variable = supporting_minarchism_var
		}
		NOT = { has_ideology = ideology:ideology_minarchist_leader }
	}

	possible = {
		custom_tooltip = {
			text = sway_to_minarchism_correct_ig_in_government_tt
			interest_group ?= {
				is_in_government = yes
				NOT = {
					is_interest_group_type = ig_trade_unions
				}
			}
		}
		custom_tooltip = {
			text = sway_to_minarchism_compatible_ideology_tt
			NOR = {
				has_ideology = ideology:ideology_integralist
				has_ideology = ideology:ideology_fascist
				has_ideology = ideology:ideology_nihilist
				has_ideology = ideology:ideology_radical
				has_ideology = ideology:ideology_social_democrat
				has_ideology = ideology:ideology_communist
				has_ideology = ideology:ideology_vanguardist
				has_ideology = ideology:ideology_anarchist
				has_ideology = ideology:ideology_humanitarian
				has_ideology = ideology:ideology_luddite
			}
		}
	}

	effect = {
		set_ideology = ideology:ideology_minarchist_leader
		owner ?= {
			add_modifier = {
				name = ir_politician_purchased_modifier
				years = 1
			}
		}
		owner = {
			purchase_politician_cost_effect = yes
		}
	}

	cooldown = { years = 1 }
	show_confirmation_box = yes

	ai_chance = {
		value = 1
	}
}

REPLACE:stts_send_paramilitaries = {
	icon = "gfx/interface/icons/ai_attitude_icons/antagonistic.dds"
	clicksound = "event:/SFX/DLC/1.3_ip1/UI/exile_character"
	show_confirmation_box = yes
	potential = {
		is_in_exile_pool = no
		is_in_void = no
		is_ruler_of_own_country = no
		is_immortal = no
		OR = {
			has_role_of_type = politician
			has_role_of_type = executive
			has_role_of_type = agitator
			has_role_of_type = magnate
		}
		owner ?= {
			exists = this
			has_technology_researched = political_agitation
		}
		is_adult = yes
	}
	show_requirements = yes
	possible = {
		owner ?= {
			OR = {
				has_law_or_variant = law_type:law_private_defense_agencies
				any_political_movement = {
					OR = {
						movement_is_socialist = yes
						is_political_movement_type = movement_anti_slavery
						is_political_movement_type = movement_pro_slavery
						is_political_movement_type = movement_radical
						is_political_movement_type = movement_fascist
					}
					political_movement_support >= 0.05
					political_movement_radicalism >= 0.10
				}
			}
		}
		OR = {
			#fascist movement will attack anyone, even other fascists
			owner ?= {
				any_political_movement = {
					is_political_movement_type = movement_fascist
					political_movement_support >= 0.05
					political_movement_radicalism >= 0.10
				}
			}
			#propertarian movement can attack anyone except non-propertarians, provided PDA is enacted
			trigger_if = {
				limit = {
					has_propertarian_ideology = no
				}
				owner ?= {
					has_law_or_variant = law_type:law_private_defense_agencies
					any_political_movement = {
						is_political_movement_type = movement_propertarian
						political_movement_support >= 0.05
						political_movement_radicalism >= 0.10
					}
				}
			}trigger_else_if = {
				limit = {
					has_socialist_ideology = no
					OR = {
						has_reactionary_ideology = yes
						has_conservative_ideology = yes
						has_liberal_ideology = yes
						has_monarchist_ideology = yes
						has_propertarian_ideology = yes
						interest_group ?= {
							is_interest_group_type = ig_industrialists
						}
						interest_group ?= {
							is_interest_group_type = ig_petty_bourgeoisie
						}
					}
				}
				owner ?= {
					any_political_movement = {
						movement_is_socialist = yes
						political_movement_support >= 0.05
						political_movement_radicalism >= 0.10
					}
				}
			}trigger_else_if = { #radicals will fight left/right
				limit = {
					has_liberal_ideology = no
					has_progressive_ideology = no
					OR = {
						has_reactionary_ideology = yes
						has_conservative_ideology = yes
						has_monarchist_ideology = yes
						has_socialist_ideology = yes
						has_propertarian_ideology = yes
					}
				}
				owner ?= {
					any_political_movement = {
						is_political_movement_type = movement_radical
						political_movement_support >= 0.05
						political_movement_radicalism >= 0.10
					}
				}
			}trigger_else_if = { #abolitionists will fight slavers and propertarians (landowners like to piggyback on propertarian movements)
				limit = {
					NOT = {
						has_ideology = ideology:ideology_abolitionist
					}
					OR = {
						has_ideology = ideology:ideology_slaver
						has_propertarian_ideology = yes
					}
				}
				owner ?= {
					any_political_movement = {
						is_political_movement_type = movement_anti_slavery
						political_movement_support >= 0.05
						political_movement_radicalism >= 0.10
					}
				}
			}trigger_else_if = { #slavers will fight abolitionists
				limit = {
					NOT = {
						has_ideology = ideology:ideology_slaver
					}
					has_ideology = ideology:ideology_abolitionist
				}
				owner ?= {
					any_political_movement = {
						is_political_movement_type = movement_pro_slavery
						political_movement_support >= 0.05
						political_movement_radicalism >= 0.10
					}
				}
			}
		}

	}

	cooldown = {
		years = 2
	}

	effect = {
		save_scope_as = relevant_character
		owner ?= {
			add_modifier = {
				name = stts_climate_of_political_violence
				years = 5
			}
		}
		random_list = {
			15 = {
				#fights off
				add_modifier = {
					name = stts_fought_off_attackers
					years = 5
				}
				interest_group ?= {
					add_modifier = {
						name = stts_galvanized_by_paramilitaries
						years = 5
					}
				}
				owner ?= {
					post_notification = stts_paramilitaries_fought_off
				}
			}
			25 = {
				#escape unharmed
				custom_tooltip = paramilitaries_fail
				owner ?= {
					post_notification = stts_paramilitaries_escaped
				}
			}
			20 = {
				#wounded
				add_trait = wounded
				owner ?= {
					post_notification = stts_paramilitaries_wounded
				}
			}
			20 = {
				#exiled
				exile_character_with_martyrdom = yes
				owner ?= {
					add_radicals = {
						value = medium_radicals
						interest_group = prev.interest_group
					}
				}
				owner ?= {
					post_notification = stts_paramilitaries_exiled
				}
			}
			20 = {
				#killed
				kill_character = yes
				interest_group ?= {
					add_modifier = {
						name = stts_galvanized_by_paramilitaries
						years = 5
					}
				}
				owner ?= {
					add_radicals = {
						value = large_radicals
						interest_group = prev.interest_group
					}
				}
				owner ?= {
					post_notification = stts_paramilitaries_killed
				}
			}
		}
	}

	ai_chance = {
		value = 0
	}
}

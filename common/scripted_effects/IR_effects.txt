ir_defense_contractors_evolution_effect = {
	ig:ig_armed_forces ?= {
		set_interest_group_name = ig_defense_contractors
		if = {
			limit = {
				has_ideology = ideology:ideology_loyalist
			}
			remove_ideology = ideology_loyalist
		}
		if = {
			limit = {
				has_ideology = ideology:ideology_loyalist_2
			}
			remove_ideology = ideology_loyalist_2
		}
		if = {
			limit = {
				has_ideology = ideology:ideology_jingoist
			}
			remove_ideology = ideology_jingoist
		}
		if = {
			limit = {
				has_ideology = ideology:ideology_patriotic
			}
			remove_ideology = ideology_patriotic
		}
		if = {
			limit = {
				has_ideology = ideology:ideology_bakufu
			}
			remove_ideology = ideology_bakufu
		}
		if = {
			limit = {
				has_ideology = ideology:ideology_sikh_militancy
			}
			remove_ideology = ideology_sikh_militancy
		}
		add_ideology = ideology_martial_corporatist
	}
}

ir_defense_contractors_reversal_effect = {
	ig:ig_armed_forces ?= {
		set_interest_group_name = ig_armed_forces
		if = {
			limit = {
				owner = {
					has_variable = human_rights_researched
				}
			}
			add_ideology = ideology_loyalist_2
		}else = {
			add_ideology = ideology_loyalist
		}
		add_ideology = ideology_jingoist
		add_ideology = ideology_patriotic
		remove_ideology = ideology_martial_corporatist
	}
}

fix_special_laws_effect = {
	if = {
		limit = { has_law_or_variant = law_type:law_directorate }
		if = {
			limit = {
				NOR = {
					has_law = law_type:law_executive_absolutism
					has_law = law_type:law_scientific_management
					has_law = law_type:law_oligopoly
					has_law = law_type:law_managed_democracy
					has_law = law_type:law_polycentric_law
				}
			}
			activate_law = law_type:law_oligopoly
		}
		if = {
			limit = {
				NOR = {
					has_law = law_type:law_voluntary_association
					has_law = law_type:law_internal_arbitration
					has_law = law_type:law_corporate_peonage
					has_law = law_type:law_equity_compensation
				}
			}
			activate_law = law_type:law_voluntary_association
		}
		if = {
			limit = { has_law = law_type:law_sovereign_monopoly }
			if = {
				limit = {
					NOR = {
						has_law = law_type:law_scientific_management
						has_law = law_type:law_executive_absolutism
					}
				}
				activate_law = law_type:law_interventionism
				if = {
					limit = {
						OR = {
							has_law = law_type:law_corporate_peonage
							has_law = law_type:law_equity_compensation
						}
					}
					if = {
						limit = { has_law = law_type:law_polycentric_law }
						activate_law = law_type:law_voluntary_association
					}
					else = { activate_law = law_type:law_internal_arbitration }
				}
			}
			else_if = {
				limit = { has_law = law_type:law_voluntary_association }
				activate_law = law_type:law_internal_arbitration
			}
		}
		else_if = {
			limit = {
				OR = {
					has_law = law_type:law_corporate_peonage
					has_law = law_type:law_equity_compensation
				}
			}
			if = {
				limit = { has_law = law_type:law_polycentric_law }
				activate_law = law_type:law_voluntary_association
			}
			else = { activate_law = law_type:law_internal_arbitration }
		}
		if = {
			limit = {
				has_law = law_type:law_internal_arbitration
				has_law = law_type:law_polycentric_law
			}
			activate_law = law_type:law_voluntary_association
		}
	}
	else = {
		if = {
			limit = { has_law = law_type:law_sovereign_monopoly }
			activate_law = law_type:law_interventionism
		}
		if = {
			limit = {
				OR = {
					has_law = law_type:law_executive_absolutism
					has_law = law_type:law_scientific_management
					has_law = law_type:law_oligopoly
					has_law = law_type:law_managed_democracy
					has_law = law_type:law_polycentric_law
				}
			}
			if = {
				limit = {
					has_law = law_type:law_polycentric_law
					has_law_or_variant = law_type:law_council_republic
				}
				activate_law = law_type:law_anarchy
			}
			else = { activate_law = law_type:law_census_voting }
		}
		if = {
			limit = {
				OR = {
					has_law = law_type:law_voluntary_association
					has_law = law_type:law_internal_arbitration
					has_law = law_type:law_corporate_peonage
					has_law = law_type:law_equity_compensation
				}
			}
			activate_law = law_type:law_right_to_associate
		}
		remove_modifier = ir_free_market_commons_modifier
	}
}

REPLACE:handle_possible_absent_caciquismo_variables = {
	if = {
		limit = {
			owner = {
				NOR = {
					has_law_or_variant = law_type:law_directorate
					has_law = law_type:law_managed_democracy
					has_law = law_type:law_oligopoly
				}
			}
		}
		if = {
			limit = {
				active_law:lawgroup_distribution_of_power ?= {
					has_amendment = amendment_type:amendment_electoral_clientelism
				}
				NOT = {
					has_variable = caciquismo_var
				}
			}
			set_variable = caciquismo_var
		}
		if = {
			limit = {
				modifier:country_forbid_electoral_fraud_bool = no
				NOT = {
					active_law:lawgroup_distribution_of_power ?= {
						has_amendment = amendment_type:amendment_electoral_clientelism
					}
				}
				NOT = {
					has_variable = electoral_fraud_abolish_var
				}
			}
			set_variable = { name = electoral_fraud_abolish_var value = 0 }
		}
	}
	else = {
		remove_variable = caciquismo_var
		remove_variable = electoral_fraud_abolish_var
		remove_variable = free_elections_var
	}
}

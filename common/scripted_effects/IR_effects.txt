ir_defense_contractors_evolution_effect = {
	ig:ig_armed_forces ?= {
		set_interest_group_name = ig_defense_contractors
		set_ig_trait = ig_trait:ig_trait_lethal_incentives

		set_ig_trait = ig_trait:ig_trait_predatory_contracting

			limit = {
				has_ideology = ideology:ideology_loyalist
			}
			remove_ideology = ideology_loyalist
		}
		if = {
			limit = {
				has_ideology = ideology:ideology_loyalist_2
			}
			remove_ideology = ideology_loyalist_2
		}
		if = {
			limit = {
				has_ideology = ideology:ideology_jingoist
			}
			remove_ideology = ideology_jingoist
		}
		if = {
			limit = {
				has_ideology = ideology:ideology_patriotic
			}
			remove_ideology = ideology_patriotic
		}
		if = {

ir_defense_contractors_reversal_effect = {
	ig:ig_armed_forces ?= {
		set_interest_group_name = ig_armed_forces
		if = {
			limit = {
				owner = {
					has_variable = human_rights_researched
				}
			}
			add_ideology = ideology_loyalist_2
		}else = {
			add_ideology = ideology_loyalist
		}
		set_ig_trait = ig_trait:ig_trait_materiel_waste
		set_ig_trait = ig_trait:ig_trait_veteran_consultation
		set_ig_trait = ig_trait:ig_trait_patriotic_fervor
		add_ideology = ideology_jingoist
		add_ideology = ideology_patriotic
		remove_ideology = ideology_martial_corporatist
	}
}

fix_special_laws_effect = {
	if = {
		limit = { has_law_or_variant = law_type:law_directorate }
		if = {
			limit = {
				NOR = {
					has_law = law_type:law_executive_absolutism
					has_law = law_type:law_scientific_management
					has_law = law_type:law_oligopoly
					has_law = law_type:law_managed_democracy
					has_law = law_type:law_polycentric_law
				}
			}
			activate_law = law_type:law_oligopoly
		}
		if = {
			limit = {
				NOR = {
					has_law = law_type:law_voluntary_association
					has_law = law_type:law_internal_arbitration
					has_law = law_type:law_corporate_peonage
					has_law = law_type:law_equity_compensation
				}
			}
			activate_law = law_type:law_voluntary_association
		}
		if = {
			limit = { has_law = law_type:law_sovereign_monopoly }
			if = {
				limit = {
					NOR = {
						has_law = law_type:law_scientific_management
						has_law = law_type:law_executive_absolutism
					}
				}
				activate_law = law_type:law_interventionism
				if = {
					limit = {
						OR = {
							has_law = law_type:law_corporate_peonage
							has_law = law_type:law_equity_compensation
						}
					}
					if = {
						limit = { has_law = law_type:law_polycentric_law }
						activate_law = law_type:law_voluntary_association
					}
					else = { activate_law = law_type:law_internal_arbitration }
				}
			}
			else_if = {
				limit = { has_law = law_type:law_voluntary_association }
				activate_law = law_type:law_internal_arbitration
			}
		}
		else_if = {
			limit = {
				OR = {
					has_law = law_type:law_corporate_peonage
					has_law = law_type:law_equity_compensation
				}
			}
			if = {
				limit = { has_law = law_type:law_polycentric_law }
				activate_law = law_type:law_voluntary_association
			}
			else = { activate_law = law_type:law_internal_arbitration }
		}
		if = {
			limit = {
				has_law = law_type:law_internal_arbitration
				has_law = law_type:law_polycentric_law
			}
			activate_law = law_type:law_voluntary_association
		}
	}
	else = {
		if = {
			limit = { has_law = law_type:law_sovereign_monopoly }
			activate_law = law_type:law_interventionism
		}
		if = {
			limit = {
				OR = {
					has_law = law_type:law_executive_absolutism
					has_law = law_type:law_scientific_management
					has_law = law_type:law_oligopoly
					has_law = law_type:law_managed_democracy
					has_law = law_type:law_polycentric_law
				}
			}
			if = {
				limit = {
					has_law = law_type:law_polycentric_law
					has_law_or_variant = law_type:law_council_republic
				}
				activate_law = law_type:law_anarchy
			}
			else = { activate_law = law_type:law_census_voting }
		}
		if = {
			limit = {
				OR = {
					has_law = law_type:law_voluntary_association
					has_law = law_type:law_internal_arbitration
					has_law = law_type:law_corporate_peonage
					has_law = law_type:law_equity_compensation
				}
			}
			activate_law = law_type:law_right_to_associate
		}
		remove_modifier = ir_free_market_commons_modifier
	}
}

REPLACE:handle_possible_absent_caciquismo_variables = {
	if = {
		limit = {
			owner = {
				NOR = {
					has_law_or_variant = law_type:law_directorate
					has_law = law_type:law_managed_democracy
					has_law = law_type:law_oligopoly
				}
			}
		}
		if = {
			limit = {
				active_law:lawgroup_distribution_of_power ?= {
					has_amendment = amendment_type:amendment_electoral_clientelism
				}
				NOT = {
					has_variable = caciquismo_var
				}
			}
			set_variable = caciquismo_var
		}
		if = {
			limit = {
				modifier:country_forbid_electoral_fraud_bool = no
				NOT = {
					active_law:lawgroup_distribution_of_power ?= {
						has_amendment = amendment_type:amendment_electoral_clientelism
					}
				}
				NOT = {
					has_variable = electoral_fraud_abolish_var
				}
			}
			set_variable = { name = electoral_fraud_abolish_var value = 0 }
		}
	}
	else = {
		remove_variable = caciquismo_var
		remove_variable = electoral_fraud_abolish_var
		remove_variable = free_elections_var
	}
}

abolish_directorate_effect = {
	if = {
		limit = { has_law = law_type:law_executive_absolutism }
		activate_law = law_type:law_autocracy
	}
	else_if = {
		limit = { has_law = law_type:law_scientific_management }
		activate_law = law_type:law_technocracy
	}
	else_if = {
		limit = { has_law = law_type:law_oligopoly }
		activate_law = law_type:law_oligarchy
	}
	else_if = {
		limit = { has_law = law_type:law_managed_democracy }
		activate_law = law_type:law_wealth_voting
	}
	else_if = {
		limit = { has_law = law_type:law_polycentric_law }
		activate_law = law_type:law_anarchy
		activate_law = law_type:law_right_to_associate
	}
	if = {
		limit = {
			OR = {
			    has_law = law_type:law_internal_arbitration
				has_law = law_type:law_corporate_peonage
			}
		}
		activate_law = law_type:law_corporatized_unions
	}
	else_if = {
		limit = { has_law = law_type:law_equity_compensation }
		activate_law = law_type:law_right_to_associate
	}
	if = {
		limit = { has_law = law_type:law_sovereign_monopoly	}
		activate_law = law_type:law_command_economy
	}
	if = {
		limit = { has_modifier = ir_free_market_commons_modifier }
		remove_modifier = ir_free_market_commons_modifier
	}
	else_if = {
		limit = { has_modifier = ir_market_commons_modifier }
		remove_modifier = ir_market_commons_modifier
	}
	if = {
		limit = { has_modifier = ir_guilded_charter_modifier}
		remove_modifier = ir_guilded_charter_modifier
	}
	else_if = {
		limit = { has_modifier = ir_production_of_security_modifier }
		remove_modifier = ir_production_of_security_modifier
	}
	else_if = {
		limit = { has_modifier = ir_monopolized_integration_modifier }
		remove_modifier = ir_monopolized_integration_modifier
	}
}

remove_incompatible_directorate_laws_effect = {
	if = {
		limit = {
			OR = {
				has_law_or_variant = law_type:law_command_economy
				has_law_or_variant = law_type:law_cooperative_ownership
			}
		}
		activate_law = law_type:law_interventionism
	}
	if = {
		limit = {
			has_law_or_variant = law_type:law_collectivized_agriculture
		}
		if = {
			limit = {
				capital ?= {
					is_in_geographic_region = geographic_region_old_world
				}
			}
			activate_law = law_type:law_peasant_proprietorship
		}else = {
			activate_law = law_type:law_homesteading
		}
	}
	if = {
		limit = {
			has_law_or_variant = law_type:law_protected_speech
		}
		activate_law = law_type:law_right_of_assembly
	}
}

check_liquidation_bypass_effect = {
	if = {
		limit = {
			NOT = { has_law_or_variant = law_type:law_guaranteed_liberties }
			NOT = { has_law = law_type:law_public_health_insurance }
			OR = {
				has_law_or_variant = law_type:law_private_schools
				has_law_or_variant = law_type:law_no_schools
			}
			OR = {
				has_law_or_variant = law_type:law_poor_laws
				has_law_or_variant = law_type:law_no_social_security
			}
		}
		custom_tooltip = public_liquidation_bypassed_tt
		if = {
			limit = {
				ig:ig_intelligentsia = {
					has_ideology = ideology:ideology_republican
				}
			}
			ig:ig_intelligentsia = {
				remove_ideology = ideology_republican
				add_ideology = ideology_market_rationalist
			}
		}
		else_if = {
			limit = {
				ig:ig_intelligentsia = {
					has_ideology = ideology:ideology_constitutionalist
				}
			}
			ig:ig_intelligentsia = {
				remove_ideology = ideology_constitutionalist
				add_ideology = ideology_market_rationalist
			}
		}
		add_modifier = {
			name = ir_free_market_commons_modifier
		}
		set_variable = public_liquidation_completed
	}
	else = {
		custom_tooltip = public_liquidation_added_tt
	}
}

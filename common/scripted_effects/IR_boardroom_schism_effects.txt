

boardroom_schism_rewards_effect = {
    if = {
        limit = {
            has_variable = minarchists_victorious
        }
        add_modifier = {
            name = ir_guilded_charter_modifier
            years = 10
        }
    }
    if = {
        limit = {
            has_variable = molinarists_victorious
        }
        add_modifier = {
            name = ir_privatized_peace_modifier
            years = 10
        }
    }
    if = {
        limit = {
            has_variable = monopolists_victorious
        }
        add_modifier = {
            name = ir_monopolized_integration_modifier
            years = 10
        }
    }
}

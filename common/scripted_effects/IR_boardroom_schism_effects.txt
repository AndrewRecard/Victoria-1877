## agitator spawning lotteries for boardroom_schism.1 ##

spawn_molinarist_agitator_effect = {
	if = {
		limit = {
			NOT = {
				any_scope_character = {
					has_ideology = ideology:ideology_molinarist_leader
					OR = {
						is_interest_group_leader = yes
						has_role_of_type = agitator
					}
				}
			}
		}
		random_interest_group = {
			limit = {
				ig_counts_as_marginal = no
			}
			weight = {
				factor = 0
				modifier = {
					factor = 2
					trigger = {
						is_interest_group_type = ig_industrialists
					}
				}
				modifier = {
					factor = 4
					trigger = {
						is_interest_group_type = ig_armed_forces
						has_ideology = ideology:ideology_martial_corporatist
					}
				}
				modifier = {
					factor = 5
					trigger = {
						is_interest_group_type = ig_rural_folk
					}
				}
				modifier = {
					factor = 5
					trigger = {
						is_interest_group_type = ig_intelligentsia
					}
				}
			}
			save_scope_as = agitator_lottery
		}
		create_character = {
			is_agitator = yes
			ideology = ideology_molinarist_leader
			interest_group = scope:agitator_lottery
		}
	}
}

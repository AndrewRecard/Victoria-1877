promote_ideology_effect = {
	remove_variable = supporting_monopolism_var
	remove_variable = supporting_minarchism_var
	remove_variable = supporting_molinarism_var
	set_variable = supporting_$ism$_var
	set_variable = {
		name = support_selection_cooldown
		value = 12
	}
	custom_tooltip = supporting_$ism$_tt
	custom_tooltip = supporting_$ism$_2_tt
	custom_tooltip = {
		text = $ism$_progress_increase_10_tt
		je:je_boardroom_schism = {
			add_progress = { value = 10 name = schism_$ism$_bar }
		}
	}
	custom_tooltip = support_selection_cooldown_tt
	if = {
		limit = {
			has_modifier = ir_suspended_lobbying_modifier
		}
		remove_modifier = ir_suspended_lobbying_modifier
	}
	if = {
		limit = { has_variable = paused_lobbying_var }
		remove_variable = paused_lobbying_var
		set_variable = {
			name = resume_lobby_cooldown
			value = 12
		}
	}
}

## agitator spawning lotteries for boardroom_schism.1 ##

spawn_molinarist_agitator_effect = {
	if = {
		limit = {
			NOT = {
				any_scope_character = {
					has_ideology = ideology:ideology_molinarist_leader
					OR = {
						is_interest_group_leader = yes
						has_role_of_type = agitator
					}
				}
			}
		}
		random_interest_group = {
			limit = {
				ig_counts_as_marginal = no
			}
			weight = {
				factor = 0
				modifier = {
					factor = 2
					trigger = {
						is_interest_group_type = ig_industrialists
					}
				}
				modifier = {
					factor = 4
					trigger = {
						is_defense_contractors = yes
					}
				}
				modifier = {
					factor = 5
					trigger = {
						is_interest_group_type = ig_rural_folk
					}
				}
				modifier = {
					factor = 5
					trigger = {
						is_interest_group_type = ig_intelligentsia
					}
				}
			}
			save_scope_as = agitator_lottery
		}
		create_character = {
			is_agitator = yes
			ideology = ideology_molinarist_leader
			interest_group = scope:agitator_lottery
		}
	}
}

        text = $boosted$_progress_increase_100_tt
        je:je_boardroom_schism = {
            add_progress = { value = 100 name = schism_$boosted$_bar }
        }
    }
    custom_tooltip = {
        text = $other1$_progress_decrease_25_tt
        je:je_boardroom_schism = {

        text = $other2$_progress_decrease_25_tt
        je:je_boardroom_schism = {

        }
    }
}

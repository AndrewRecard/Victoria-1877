je_monopolist_mandate = {
	icon = "gfx/interface/icons/invention_icons/corporate_charters.dds"
	group = je_group_propertarianism
	should_be_pinned_by_default_uninvolved_or_context = yes
	weight = 100

	is_shown_when_inactive = {
		always = no
	}

	immediate = {
		trigger_event = {
			id = boardroom_schism.4
			popup = yes
		}
	}

	complete = {
		has_law = law_type:law_sovereign_monopoly
	}
	on_complete = {
		set_variable = monopolists_victorious
		add_modifier = {
			name = ir_monopolized_integration_modifier
		}
		add_modifier = {
			name = ir_schism_resolved_modifier
			days = normal_modifier_time
		}
		add_loyalists = {
			value = large_radicals
		}
		remove_modifier = ir_schism_mandate_modifier
	}

	invalid = {
		NOT = {
			has_journal_entry = je_boardroom_schism
		}
	}
	on_invalid = {
		remove_modifier = ir_schism_mandate_modifier
	}

	timeout = short_modifier_time
	on_timeout = {
		custom_tooltip = "je_schism_fail"
		random_political_movement = {
			limit = {
				is_political_movement_type = movement_propertarian
			}
			set_core_ideology = ideology_propertarian_movement
		}
		set_variable = mandate_wasted
		remove_modifier = ir_schism_mandate_modifier
	}
}

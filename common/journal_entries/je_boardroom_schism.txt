je_boardroom_schism = {
	icon = "gfx/interface/icons/invention_icons/corporate_management.dds"
	group = je_group_propertarianism

	scripted_progress_bar = schism_monopolist_bar
	scripted_progress_bar = schism_minarchist_bar
	scripted_progress_bar = schism_molinarist_bar

	scripted_button = promote_monopolism_button
	scripted_button = promote_minarchism_button
	scripted_button = promote_molinarism_button
	scripted_button = suspend_lobbying_button
	scripted_button = sponsor_thinktanks_button
	scripted_button = convene_egm_button

	is_shown_when_inactive = {
		has_technology_researched = corporate_charters
	}
	possible = {
		has_law_or_variant = law_type:law_directorate
		NOR = {
			has_law = law_type:law_polycentric_law
			has_law = law_type:law_sovereign_monopoly
		}
		has_technology_researched = corporate_management
		has_technology_researched = anarchism
		has_technology_researched = corporatism
		any_scope_politician = {
		}
		calc_true_if = {
			amount >= 2
			any_scope_politician = {
				has_ideology = ideology:ideology_minarchist_leader
			}
			any_scope_politician = {
				has_ideology = ideology:ideology_monopolist_leader
			}
			any_scope_politician = {
				has_ideology = ideology:ideology_molinarist_leader
			}
		}
		custom_tooltip = {
			text = completed_public_liquidation_tt
			has_variable = public_liquidation_completed
		}
	}

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = blank_tt
				trigger = {
					NOT = {
						has_journal_entry = je_boardroom_schism
					}
				}
			}
			triggered_desc = { # Cementing the Rightful Dynasty is active
				desc = status_monopolist_mandate
				trigger = {
					has_journal_entry = je_monopolist_mandate
				}
			}
			triggered_desc = { # Cementing the Rightful Dynasty is active
				desc = status_molinarist_mandate
				trigger = {
					has_journal_entry = je_molinarist_mandate
				}
			}
			triggered_desc = {
				desc = status_monopolism_winning
				trigger = {
					scope:journal_entry = {
						"scripted_bar_progress(schism_monopolist_bar)" > "scripted_bar_progress(schism_minarchist_bar)"
						"scripted_bar_progress(schism_monopolist_bar)" > "scripted_bar_progress(schism_molinarist_bar)"
					}
				}
			}
			triggered_desc = {
				desc = status_molinarism_winning
				trigger = {
					scope:journal_entry = {
						"scripted_bar_progress(schism_molinarist_bar)" > "scripted_bar_progress(schism_minarchist_bar)"
						"scripted_bar_progress(schism_molinarist_bar)" > "scripted_bar_progress(schism_monopolist_bar)"
					}
				}
			}
			triggered_desc = {
				desc = status_minarchism_winning
				trigger = {
					scope:journal_entry = {
						"scripted_bar_progress(schism_minarchist_bar)" > "scripted_bar_progress(schism_molinarist_bar)"
						"scripted_bar_progress(schism_minarchist_bar)" > "scripted_bar_progress(schism_monopolist_bar)"
					}
				}
			}
			triggered_desc = {
				desc = status_no_winner
				trigger = {
					always = yes
				}
			}
		}
		triggered_desc = {
			desc = je_boardroom_lobbying_cost
			trigger = {
				has_journal_entry = je_boardroom_schism
				NOT = {
					has_variable = boardroom_schism_freeze
					has_variable = suspended_lobbying_var
				}
			}
		}
	}

	modifiers_while_active = {
		ir_propertarian_lobby_war_modifier
	}

	immediate = {
		trigger_event = {
			id = boardroom_schism.1
			popup = yes
		}
		random_law = {
			limit = { type = law_type:law_directorate }
			save_scope_as = directorate_scope
		}
		set_variable = {
			name = minarchist_effects_var
			value = 0
		}
		set_variable = {
			name = molinarist_effects_var
			value = 0
		}
		set_variable = {
			name = monopolist_effects_var
			value = 0
		}
	}

	on_weekly_pulse = {
		effect = {
			remove_modifier = ir_funding_lobbies_modifier
			add_modifier = {
				name = ir_funding_lobbies_modifier
				multiplier = lobbying_expenditure_value
			}
			if = {
				limit = {
					scope:journal_entry = {
						"scripted_bar_progress(schism_molinarist_bar)" >= 400
					}
					NOT = {
						has_variable = boardroom_schism_freeze
					}
				}
				set_variable = boardroom_schism_freeze
				add_journal_entry = {
					type = je_molinarist_mandate
				}
			}
			if = {
				limit = {
					scope:journal_entry = {
						"scripted_bar_progress(schism_monopolist_bar)" >= 400
					}
					NOT = {
						has_variable = boardroom_schism_freeze
					}
				}
				set_variable = boardroom_schism_freeze
				add_journal_entry = {
					type = je_monopolist_mandate
				}
			}
			if = {
				limit = {
					scope:journal_entry = {
						"scripted_bar_progress(schism_minarchist_bar)" >= 400
					}
					NOT = {
						has_variable = boardroom_schism_freeze
					}
				}
				set_variable = boardroom_schism_freeze
				trigger_event = {
					id = boardroom_schism.6
					popup = yes
				}
			}
		}
	}

	on_monthly_pulse = {
		random_events = {
			15 = 0
			1 = boardroom_schism.7
		}
	}

	complete = {
		has_law = law_type:law_directorate
		custom_tooltip = {
			text = je_schism_complete
			OR = {
				has_variable = monopolists_victorious
				has_variable = minarchists_victorious
				has_variable = molinarists_victorious
			}
		}
	}
	on_complete = {
		custom_tooltip = {
			text = je_schism_resulting_je
			clear_boardroom_schism_variables = yes
		}
	}

	fail = {
		OR = {
		    custom_tooltip = {
    			text = "je_schism_no_mandate"
    			has_variable = mandate_wasted
    		}
    		custom_tooltip = {
				always = no
    			text = "je_schism_early_laws"
    		}
		}
	}

	on_fail = {
		if = {
			limit = { has_variable = mandate_wasted }
		}
		add_modifier = {
			name = ir_schism_failed_modifier
			days = long_modifier_time
		}
		clear_boardroom_schism_variables = yes
	}

	invalid = {
		NOT = {
			has_law = law_type:law_directorate
		}
		OR = {
			has_law = law_type:law_polycentric_law
			has_law = law_type:law_sovereign_monopoly
		}
	}
	on_invalid = {
		clear_boardroom_schism_variables = yes
	}

	weight = 100
	should_be_pinned_by_default = yes
}

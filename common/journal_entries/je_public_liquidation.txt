#Public insolvency crisis
je_public_liquidation = {
	icon = "gfx/interface/icons/invention_icons/elevator.dds"
	group = je_group_propertarianism
	should_be_pinned_by_default = yes

	is_shown_when_inactive = {
		has_technology_researched = corporate_charters
		NOR = {
			has_modifier = ir_free_market_commons_modifier
			has_modifier = ir_market_commons_modifier
		}
	}

	possible = {
		has_law_or_variant = law_type:law_directorate
	}

	modifiers_while_active = {
		ir_austerity_measures_modifier
		ir_public_divestiture_modifier
	}

	on_monthly_pulse = {
		random_events = {
			1 = 0
			100 = directorate_events.3
			100 = directorate_events.4
			100 = directorate_events.5
		}
	}

	immediate = {
	}

	scripted_progress_bar = ir_insolvency_bar

	complete = {
		has_law = law_type:law_private_defense_agencies
		NOT = {
			has_law = law_type:law_public_health_insurance
		}
		OR = {
			has_law = law_type:law_private_schools
			has_law = law_type:law_no_schools
		}
		OR = {
			has_law = law_type:law_poor_laws
			has_law = law_type:law_no_social_security
		}
	}

	on_complete = {
		add_modifier = {
			name = ir_free_market_commons_modifier
		}
		add_modifier = {
			name = ir_institutional_redress_modifier
			years = 5
		}
		if = {
			limit = {
				ig:ig_intelligentsia = {
					has_ideology = ideology:ideology_republican
				}
			}
			ig:ig_intelligentsia = {
				remove_ideology = ideology_republican
				add_ideology = ideology_market_rationalist
			}
		}else_if = {
			limit = {
				ig:ig_intelligentsia = {
					has_ideology = ideology:ideology_constitutionalist
				}
			}
			ig:ig_intelligentsia = {
				remove_ideology = ideology_constitutionalist
				add_ideology = ideology_market_rationalist
			}
		}
		custom_tooltip = {
			text = propertarian_split_effect_tt
			propertarian_split_effect = yes
		}
		set_variable = public_liquidation_completed
	}

	fail = {
		OR = {
			has_modifier = declared_bankruptcy
			custom_tooltip = {
				text = ir_insolvency_bar_full
				scope:journal_entry = {
					"scripted_bar_progress(ir_insolvency_bar)" >= 1000
				}
			}
		}
	}

	on_fail = {
		custom_tooltip = {
			text = institutional_collapse_tt
			trigger_event = {
				id = directorate_events.8
				popup = yes
			}
		}
		custom_tooltip = {
			text = propertarian_split_effect_tt
			propertarian_split_effect = yes
		}
		set_variable = public_liquidation_completed
	}

	invalid = {
		NOT = {
			has_law_or_variant = law_type:law_directorate
		}
	}
}

#The state military is dismantled in favour of private defense agencies
je_defense_incorporation = {
	icon = "gfx/interface/icons/invention_icons/power_of_the_purse.dds"
	group = je_group_propertarianism
	should_be_pinned_by_default_uninvolved_or_context = yes

	is_shown_when_inactive = {
		has_technology_researched = general_staff
		has_defense_contractors = no
	}

	scripted_progress_bar = ir_defense_incorporation_bar

	widget = {
		gui = "gui/journal_entry_widgets/IR_defense_incorporation_widgets.gui"
		name = "widget_je_defense_incorporation_leader"
		container = "custom_widget_container_je_icon"
	}

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = blank_tt
				trigger = {
					NOT = {
						has_journal_entry = je_defense_incorporation
					}
				}
			}
			triggered_desc = {
				desc = market_liberal_armed_forces_leader_tt
				trigger = {
					any_interest_group = {
						is_interest_group_type = ig_armed_forces
						leader = {
							OR = {
								has_propertarian_ideology = yes
								has_ideology = ideology:ideology_market_liberal
							}
						}
					}
				}
			}
			triggered_desc = {
				desc = no_market_liberal_armed_forces_leader_tt
				trigger = {
					any_interest_group = {
						is_interest_group_type = ig_armed_forces
						leader = {
							NOR = {
								has_propertarian_ideology = yes
								has_ideology = ideology:ideology_market_liberal
							}
						}
					}
				}
			}
		}
	}

	scripted_button = repeal_defense_agencies_button
	scripted_button = depose_armed_forces_leader_button
	
	on_monthly_pulse = {
		random_events = {
			22 = 0
			1 = defense_incorporation.2
			1 = defense_incorporation.3
			1 = defense_incorporation.4
			1 = defense_incorporation.5
		}
	}

	possible = {
		has_law_or_variant = law_type:law_private_defense_agencies
	}
	
	immediate = {
		ig:ig_armed_forces = {
			leader = {
				save_scope_as = af_leader_scope
			}
		}
		set_variable = {
			name = af_leader_var
			value = scope:af_leader_scope
		}
		trigger_event = defense_incorporation.1
	}

	on_weekly_pulse = {
		effect = {
			ig:ig_armed_forces = {
				leader = {
					save_scope_as = af_leader_refresh
					if = {
						limit = {
							NOT = {
								has_modifier = ir_leading_restructure_modifier
							}
						}
						add_modifier = {
							name = ir_leading_restructure_modifier
						}
					}
				}
			}
			set_variable = {
				name = af_leader_var
				value = scope:af_leader_refresh
			}
		}
	}

	modifiers_while_active = {
		ir_defense_agencies_reform_modifier
	}
	
	complete = {
		ig:ig_armed_forces = {
			ig_counts_as_marginal = no
		}
		custom_tooltip = {
			text = progress_bar_complete
			scope:journal_entry = {
				"scripted_bar_progress(ir_defense_incorporation_bar)" >= 100
			}
		}
	}

	on_complete = {
		trigger_event = {
			id = defense_incorporation.6
			popup = yes
		}
		remove_variable = af_leader_var
		hidden_effect = {
			if = {
				limit = {
					any_scope_character = {
						has_modifier = ir_leading_restructure_modifier
					}
				}
				every_scope_character = {
					remove_modifier = ir_leading_restructure_modifier
				}
			}
		}
	}

	invalid = {
		NOT = {
			has_law_or_variant = law_type:law_private_defense_agencies
		}
	}

	on_invalid = {
		remove_variable = af_leader_var
		hidden_effect = {
			if = {
				limit = {
					any_scope_character = {
						has_modifier = ir_leading_restructure_modifier
					}
				}
				every_scope_character = {
					remove_modifier = ir_leading_restructure_modifier
				}
			}
		}
	}
}

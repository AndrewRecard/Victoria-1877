on_yearly_pulse_country = {
    on_actions = {
        entrenched_entitlements_timer_effect
    }
}

on_acquired_technology = {
    events = {
        propertarian_origins.1
    }
}

entrenched_entitlements_timer_effect = {
    effect = {
        if = {
            limit = {
                NOT = {
                    has_variable = entrenched_entitlements_var
                }
                has_variable = entrenched_entitlement_years
            }
            change_variable = {
                name = entrenched_entitlement_years
                add = 1
            }
            if = {
                limit = {
                    var:entrenched_entitlement_years = 10
                }
                trigger_event = { id = amendment_events.1 }
            } else_if = {
                limit = { var:entrenched_entitlement_years = 15 }
                trigger_event = { id = amendment_events.2 }
            }
        } else_if = {
            limit = {
                has_variable = entrenched_entitlements_var
            }
            remove_variable = entrenched_entitlement_years
        }
    }
}

on_monthly_pulse_country = {
    on_actions = {
        franchise_investment_efficiency_on_action
    }
}

franchise_investment_efficiency_on_action = {
    effect = {
        remove_modifier = ir_franchising_level_1_modifier
        remove_modifier = ir_franchising_level_2_modifier
        remove_modifier = ir_franchising_level_3_modifier

        if = {
            limit = {
                is_power_bloc_leader = yes
                power_bloc = { has_principle = principle:principle_foreign_investment_1 }
            }
            add_modifier = {
                name = ir_franchising_level_1_modifier
                multiplier = franchise_directorate_count_value
            }
        }
        else_if = {
            limit = {
                is_power_bloc_leader = yes
                power_bloc = { has_principle = principle:principle_foreign_investment_2 }
            }
            add_modifier = {
                name = ir_franchising_level_2_modifier
                multiplier = franchise_directorate_count_value
            }
        }
        else_if = {
            limit = {
                is_power_bloc_leader = yes
                power_bloc = { has_principle = principle:principle_foreign_investment_3 }
            }
            add_modifier = {
                name = ir_franchising_level_3_modifier
                multiplier = franchise_directorate_count_value
            }
        }
    }
}

on_company_disbanded = {
    on_actions = {
        disbanded_propertarian_company_on_action
    }
}

disbanded_propertarian_company_on_action = {
    effect = {
        if = {
            limit = {
                has_journal_entry = je_boardroom_schism
            }
            if = {
                limit = {
                    scope:company = {
                        executive ?= {
                            has_ideology = ideology:ideology_minarchist_leader
                        }
                    }
                }
                every_interest_group = {
                    limit = {
                        leader = {
                            has_ideology = ideology:ideology_minarchist_leader
                        }
                    }
                    add_modifier = {
                        name = ir_disbanded_company_modifier
                        days = normal_modifier_time
                    }
                }
            }
            if = {
                limit = {
                    scope:company = {
                        executive = {
                            has_ideology = ideology:ideology_molinarist_leader
                        }
                    }
                }
                every_interest_group = {
                    limit = {
                        leader = {
                            has_ideology = ideology:ideology_molinarist_leader
                        }
                    }
                    add_modifier = {
                        name = ir_disbanded_company_modifier
                        days = normal_modifier_time
                    }
                }
            }
            if = {
                limit = {
                    scope:company = {
                        executive = {
                            has_ideology = ideology:ideology_monopolist_leader
                        }
                    }
                }
                every_interest_group = {
                    limit = {
                        leader = {
                            has_ideology = ideology:ideology_monopolist_leader
                        }
                    }
                    add_modifier = {
                        name = ir_disbanded_company_modifier
                        days = normal_modifier_time
                    }
                }
            }
        }
    }
}

on_secession_start = {
    on_actions = {
        fix_special_laws_on_action
    }
}

on_revolution_start = {
    on_actions = {
        fix_special_laws_on_action
    }
}

fix_special_laws_on_action = {
    effect = {
        scope:target = {
            fix_special_laws_effect = yes
        }
    }
}

on_half_yearly_pulse_country = {
    on_actions = {
        evaluate_police_synergy
        sovereign_monopoly_conversion_effect
    }
}

evaluate_police_synergy = {
    effect  = {
        if = {
            limit = {
                has_law_or_variant = law_type:law_private_defense_agencies
                has_law_or_variant = law_type:law_militarized_police
                NOT = {
                    has_modifier = ir_private_militarized_police
                }
            }
            add_modifier = {
                name = ir_private_militarized_police
            }
        }else_if = {
            limit = {
                has_modifier = ir_private_militarized_police
                NOT = {
                    has_law_or_variant = law_type:law_private_defense_agencies
                    has_law_or_variant = law_type:law_militarized_police
                }
            }
            remove_modifier = ir_private_militarized_police
        }
    }
}

sovereign_monopoly_conversion_effect = {
    effect = {
        if = {
            limit = {
                has_law = law_type:law_sovereign_monopoly
            }
            add_modifier = {
                name = ir_sm_throughput_modifier
                multiplier = sovereign_monopoly_throughput_conversion
                months = 6
            }
            add_modifier = {
                name = ir_sm_construction_modifier
                multiplier = sovereign_monopoly_construction_conversion
                months = 6
            }
        }
    }
}

can_promote_ideology_trigger = {
	any_interest_group = {
		is_in_government = yes
		leader = {
			has_ideology = ideology:ideology_$ist$_leader
		}
	}
	custom_tooltip = {
		text = legitimate_government_tt
		legitimacy >= 25
	}
	custom_tooltip = {
		text = not_already_supporting_$ism$
		NOT = { has_variable = supporting_$ism$_var }
	}
	custom_tooltip = {
		text = not_recently_suspended_lobbying_tt
		OR = {
			has_variable = paused_lobbying_var
			NOT = { has_variable = pause_lobby_cooldown }
		}
	}
	custom_tooltip = {
		text = support_selection_on_cooldown_tt
		OR = {
			has_variable = paused_lobbying_var
			NOT = { has_variable = support_selection_cooldown }
		}
	}
}

has_propertarian_ideology = {
	custom_tooltip = {
		text = has_propertarian_ideology_tt
		OR = {
			has_ideology = ideology:ideology_molinarist_leader
			has_ideology = ideology:ideology_minarchist_leader
			has_ideology = ideology:ideology_monopolist_leader
		}
	}
}

REPLACE:ideology_market_liberal_valid_trigger = {
	OR = {
		is_interest_group_type = ig_petty_bourgeoisie
		is_interest_group_type = ig_landowners
		is_interest_group_type = ig_intelligentsia
		is_interest_group_type = ig_industrialists
	}
	NOT = {
		owner = {
			has_law_or_variant = law_type:law_directorate
		}
	}
}

REPLACE:ideology_protectionist_valid_trigger = {
	NOR = {
		is_interest_group_type = ig_devout
		owner = {
			has_law_or_variant = law_type:law_directorate
		}
	}
}

ideology_minarchist_valid_trigger = {
	NOT = {
		is_interest_group_type = ig_trade_unions
	}
}

ideology_molinarist_valid_trigger = {
	NOR = {
		is_interest_group_type = ig_trade_unions
		is_interest_group_type = ig_landowners
	}
}

ideology_monopolist_valid_trigger = {
	NOR = {
		is_interest_group_type = ig_trade_unions
		is_interest_group_type = ig_rural_folk
	}
}

shareholders_alliance = {
	color = { 25 55 105 }

	name = {
		first_valid = {
			triggered_desc = {
				desc = shareholders_alliance
				trigger = {
					always = yes
				}
			}
		}
	}

	icon = {
		default = "gfx/interface/icons/political_parties/free_trade.dds"
	}

	# THIS IS RUN ONLY ONCE AT GAME START
	valid_for_country = {
		always = yes
	}

	available_for_interest_group = {
		owner = {
			has_law_or_variant = law_type:law_directorate
			country_has_voting_franchise = yes
			OR = {
				py:shareholders_alliance ?= {
					any_member = {}
				}
				any_interest_group = {
					ig_counts_as_marginal = no
					leader ?= {
						NOR = {
							has_ideology = ideology:ideology_molinarist_leader
							has_ideology = ideology:ideology_monopolist_leader
							has_ideology = ideology:ideology_integralist
							has_ideology = ideology:ideology_fascist
							has_ideology = ideology:ideology_nihilist
							has_ideology = ideology:ideology_radical
							has_ideology = ideology:ideology_social_democrat
							has_ideology = ideology:ideology_communist
							has_ideology = ideology:ideology_vanguardist
							has_ideology = ideology:ideology_anarchist
							has_ideology = ideology:ideology_humanitarian
							has_ideology = ideology:ideology_republican_leader
							has_ideology = ideology:ideology_authoritarian
							has_ideology = ideology:ideology_positivist
							has_ideology = ideology:ideology_royalist
							has_ideology = ideology:ideology_traditionalist
						}
					}
				}
			}
		}
	}

	# Root - IG considering joining
	# scope:number - the number of IGs that would join this party before the root IG
	join_weight = {
		value = {
			value = -5
			desc = "base_value"
		}

		if = {
			limit = {
				leader ?= {
					has_ideology = ideology:ideology_minarchist_leader
				}
			}
			add = {
				value = 20
				desc = "from_minarchist_leader"
			}
		}
		if = {
			limit = {
				leader ?= {
					OR = {
						has_ideology = ideology:ideology_moderate
						has_ideology = ideology:ideology_liberal
						has_ideology = ideology:ideology_reformer
						has_ideology = ideology:ideology_market_liberal
					}
				}
			}
			add = {
				value = 15
				desc = "from_minarchist_compatible"
			}
		}
		if = {
			limit = {
				is_interest_group_type = ig_armed_forces
				owner = {
					has_law_or_variant = law_type:law_dedicated_police
				}
			}
			add = {
				value = 15
				desc = "law_dedicated_police"
			}
		}else_if = {
			limit = {
				is_interest_group_type = ig_armed_forces
			}
			add = {
				value = 10
				desc = "ig_armed_forces"
			}
		}
		if = {
			limit = {
				is_interest_group_type = ig_petty_bourgeoisie
			}
			add = {
				value = 10
				desc = "ig_petty_bourgeoisie"
			}
		}
		if = {
			limit = {
				is_interest_group_type = ig_trade_unions
			}
			add = {
				value = -25
				desc = "ig_trade_unions"
			}
		}

		if = {
			limit = {
				is_interest_group_type = ig_industrialists
				is_powerful = yes
			}
			multiply = {
				value = 2
				desc = "from_archetype_party"
			}
		}
		if = {
			limit = {
				scope:number <= 1
				is_powerful = no
				ig_counts_as_marginal = no
			}
			multiply = {
				value = 2
				desc = "from_influential_supporting_member"
			}
		}
		else_if = {
			limit = {
				scope:number = 2
			}
			multiply = {
				value = 0.5
				desc = "from_two_prior_members"
			}
		}
		else_if = {
			limit = {
				scope:number = 3
			}
			multiply = {
				value = 0.25
				desc = "from_three_prior_members"
			}
		}
	}
}

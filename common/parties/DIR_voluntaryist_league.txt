voluntaryist_league = {
	color = { 215 160 30 }
	name = {
		first_valid = {
			triggered_desc = {
				desc = voluntaryist_league
				trigger = {
					always = yes
				}
			}
		}
	}
	icon = {
		default = "gfx/interface/icons/political_parties/anarchist.dds"
	}
	valid_for_country = {
		always = yes
	}
	available_for_interest_group = {
		owner = {
			has_law_or_variant = law_type:law_directorate
			country_has_voting_franchise = yes
			OR = {
				py:national_trust_party ?= {
					any_member = {}
				}
				any_interest_group = {
					ig_counts_as_marginal = no
					OR = {
						is_interest_group_type = ig_industrialists
						leader ?= {
							NOR = {
								has_ideology = ideology:ideology_minarchist_leader
								has_ideology = ideology:ideology_monopolist_leader
								has_ideology = ideology:ideology_radical
								has_ideology = ideology:ideology_social_democrat
								has_ideology = ideology:ideology_communist
								has_ideology = ideology:ideology_vanguardist
								has_ideology = ideology:ideology_humanitarian
								has_ideology = ideology:ideology_republican_leader
								has_ideology = ideology:ideology_royalist
								has_ideology = ideology:ideology_liberal
								has_ideology = ideology:ideology_reformer
								has_ideology = ideology:ideology_ethno_nationalist
							}
						}
					}
				}
			}
		}
	}

	join_weight = {
		value = {
			value = -5
			desc = "base_value"
		}

		if = {
			limit = {
				leader ?= {
					has_ideology = ideology:ideology_molinarist_leader
				}
			}
			add = {
				value = 25
				desc = "from_molinarist_leader"
			}
		}
		if = {
			limit = {
				leader ?= {
					OR = {
						has_ideology = ideology:ideology_anarchist
						has_ideology = ideology:ideology_market_liberal
						has_ideology = ideology:ideology_nihilist
					}
				}
			}
			add = {
				value = 15
				desc = "from_molinarist_compatible"
			}
		}
		if = {
			limit = {
				is_interest_group_type = ig_armed_forces
				owner = {
					OR = {
						has_law_or_variant = law_type:law_no_police
						has_law_or_variant = law_type:law_local_police
					}
				}
			}
			add = {
				value = 15
				desc = "no_or_local_police"
			}
		}else_if = {
			limit = {
				is_interest_group_type = ig_armed_forces
			}
			add = {
				value = 5
				desc = "ig_armed_forces"
			}
		}
		if = {
			limit = {
				is_interest_group_type = ig_intelligentsia
			}
			add = {
				value = 5
				desc = "ig_intelligentsia"
			}
		}
		if = {
			limit = {
				is_interest_group_type = ig_rural_folk
			}
			add = {
				value = 5
				desc = "ig_rural_folk"
			}
		}
		if = {
			limit = {
				is_interest_group_type = ig_trade_unions
			}
			add = {
				value = -25
				desc = "ig_trade_unions"
			}
		}
		if = {
			limit = {
				owner = {
					py:national_trust_party ?= {
						any_member = {
							count >= 3
						}
					}
					exists = py:shareholders_alliance
				}
			}
			multiply = {
				value = 0.5
				desc = "centrist_alliance_countering_ntp"
			}
		}
		if = {
			limit = {
				scope:number <= 1
				is_powerful = no
				ig_counts_as_marginal = no
			}
			multiply = {
				value = 2
				desc = "from_influential_supporting_member"
			}
		}
	}
}

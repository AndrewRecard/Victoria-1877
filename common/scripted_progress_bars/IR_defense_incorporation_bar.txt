ir_defense_incorporation_bar = {
 	name = "ir_defense_incorporation_bar"
 	desc = "ir_defense_incorporation_bar_desc"

 	default_green = yes

 	monthly_progress = {
		if = {
		    limit = {
				has_journal_entry = je_defense_incorporation
		    }
		    add = {
    			value = 1
				desc = monthly_progress
    		}

    		if = {
    			limit = {
    				ig:ig_armed_forces = {
    					ig_counts_as_marginal = yes
    				}
    			}
    			add = {
    				value = -0.5
    				desc = "armed_forces_marginalized"
    			}
    		}

			add = {
				desc = officers_radicals
				add = {
					every_scope_pop = {
						limit = { is_pop_type = officers }
						add = {
							value = total_size
							multiply = pop_radical_fraction
						}
					}
				}
				divide = {
					every_scope_pop = {
						limit = { is_pop_type = officers }
						add = total_size
					}
					min = 1
				}
				multiply = -1
			}

			add = {
				desc = officers_loyalists
				add = {
					every_scope_pop = {
						limit = { is_pop_type = officers }
						add = {
							value = total_size
							multiply = pop_loyalist_fraction
						}
					}
				}
				divide = {
					every_scope_pop = {
						limit = { is_pop_type = officers }
						add = total_size
					}
					min = 1
				}
				multiply = 1
			}

			add = {
    			ig:ig_armed_forces = {
    				leader = {
    					add = loyalty
    				}
    			}
    			subtract = 50
    			multiply = 0.01
    			desc = "armed_forces_leader_loyalty"
    		}

			add = {
				desc = "average_loyalty_of_generals"
				every_scope_character = {
					limit = {
						has_role_of_type = general
					}
					add = {
						add = {
							add = loyalty
							subtract = 50
						}
						multiply = prominence
					}
				}
				divide = {
					every_scope_character = {
						limit = {
							has_role_of_type = general
						}
						add = prominence
					}
				}
				multiply = 0.01
			}

			if = {
    			limit = {
    				ig:ig_armed_forces = {
    					leader = {
    						propertarian_or_market_liberal = yes
    					}
    				}
    			}
    			add = {
    				value = 1
    				desc = "propertarian_armed_forces_leader"
    			}
    		}

			if = {
				limit = {
					is_at_war = yes
				}
				add = {
					value = 1
					desc = "at_war"
				}
			}
		}
	}

 	start_value = 0
 	min_value = 0
 	max_value = 100
}

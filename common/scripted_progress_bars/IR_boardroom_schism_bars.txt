schism_minarchism_bar = {
	name = "minarchism_bar"
	desc = "minarchism_bar_desc"
	default_bad = yes
	monthly_progress = {
		if = {
			limit = {
				has_journal_entry = je_boardroom_schism
				NOR = {
					has_variable = boardroom_schism_freeze
					has_variable = paused_lobbying_var
				}
			}
			### From Interest Groups
			if = {
				limit = {
					any_interest_group = {
						leader = { has_ideology = ideology:ideology_minarchist_leader }
						ig_counts_as_marginal = no
					}
				}
				add = {
					desc = "minarchism_progress_from_igs_tt"
					every_interest_group = {
						limit = {
							leader = { has_ideology = ideology:ideology_minarchist_leader }
							ig_counts_as_marginal = no
						}
						add = this.ig_clout
					}
					multiply = 5
				}
			}
			### From Popularity
			if = {
				limit = {
					any_scope_character = {
						has_ideology = ideology:ideology_minarchist_leader
						OR = {
							has_role_of_type = politician
							has_role_of_type = executive
							has_role_of_type = agitator
							has_role_of_type = magnate
						}
					}
				}
				add = {
					desc = "minarchism_progress_from_characters_tt"
					every_scope_character = {
						limit = {
							has_ideology = ideology:ideology_minarchist_leader
							OR = {
								has_role_of_type = politician
								has_role_of_type = executive
								has_role_of_type = agitator
								has_role_of_type = magnate
							}
						}
						add = {
							add = popularity
							multiply = {
								add = prominence
								divide = 100
							}
						}
					}
					divide = 100
				}
			}
			### From Prominence of Execs and Magnates
			if = {
				limit = {
					any_scope_character = {
						has_ideology = ideology:ideology_minarchist_leader
					}
				}
				add = {
					desc = "minarchism_progress_from_prominence_tt"
					every_scope_character = {
						limit = {
							OR = {
								has_role_of_type = magnate
								has_role_of_type = executive
							}
							has_ideology = ideology:ideology_minarchist_leader
						}
						add = {
							add = prominence
							divide = 100
						}
					}
				}
			}
			### From Num Prosperous Companies
			if = {
				limit = { num_companies > 0 }
				add = {
					desc = "minarchism_progress_from_average_prosperity_tt"
					add = {
					    add = {
	    					every_company = {
	    						add = {
	    							value = company_prosperity
	    							multiply = company_employed_levels
	    						}
	    					}
	    				}
	    				divide = {
	    					every_company = {
	    						add = company_employed_levels
	    					}
	    				}
	    				multiply = 0.03
					}
					subtract = 0.5
				}
			}
			### Minarchists gain progress from legitimacy in government
			if = {
				limit = {
					any_interest_group = {
						leader = {
							has_ideology = ideology:ideology_minarchist_leader
						}
					}
				}
				add = {
					desc = "minarchism_progress_from_minarchist_legitimacy_tt"
					value = legitimacy
					multiply = 0.1
					subtract = 2.5
				}
			}
			### From Effects (event/button nudges)
			if = {
				limit = {
					exists = var:minarchist_effects_var
					NOT = { var:minarchist_effects_var = 0 }
				}
				add = {
					desc = "progress_from_events_tt"
					value = owner.var:minarchist_effects_var
				}
			}
			### From Mean Reversion
			multiply = {
				desc = "progress_from_mean_reversion_tt"
				add = {
					add = {
						je:je_boardroom_schism = { value = "scripted_bar_progress(schism_minarchism_bar)" }
					}
					add = {
						je:je_boardroom_schism = { value = "scripted_bar_progress(schism_molinarism_bar)" }
					}
					add = {
						je:je_boardroom_schism = { value = "scripted_bar_progress(schism_monopolism_bar)" }
					}
					divide = {
						add = {
							je:je_boardroom_schism = { value = "scripted_bar_progress(schism_minarchism_bar)" add = 1 }
						}
						multiply = 3
					}
				}
			}
		}
	}
	start_value = 10
	min_value = 0
	max_value = 600
}

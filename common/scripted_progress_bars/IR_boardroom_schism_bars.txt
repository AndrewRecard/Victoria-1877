schism_minarchist_bar = {
	name = "minarchism_bar"
	desc = "minarchism_bar_desc"
	default_bad = yes
	monthly_progress = {
		if = {
			limit = {
				owner = {
					NOT = { has_variable = boardroom_schism_freeze }
				}
			}
			### From Interest Groups
			if = {
				limit = {
					any_interest_group = {
						leader = { has_ideology = ideology:ideology_minarchist_leader }
						ig_counts_as_marginal = no
					}
				}
				add = {
					desc = "minarchism_progress_from_igs_tt"
					every_interest_group = {
						limit = {
							leader = { has_ideology = ideology:ideology_minarchist_leader }
							ig_counts_as_marginal = no
						}
						add = this.ig_clout
					}
					multiply = 5
				}
			}
			### From Leaders
			if = {
				limit = {
					any_scope_character = {
						has_ideology = ideology:ideology_minarchist_leader
						OR = {
							has_role = politician
							has_role = executive
							has_role = agitator
						}
					}
				}
				add = {
					desc = "minarchism_progress_from_characters_tt"
					every_scope_character = {
						limit = {
							has_ideology = ideology:ideology_minarchist_leader
							OR = {
								has_role = politician
								has_role = executive
								has_role = agitator
							}
						}
						add = this.popularity
					}
					divide = 100
				}
			}
			### From Executives
			if = {
				limit = {
					any_scope_character = {
						has_role = executive
						has_ideology = ideology:ideology_minarchist_leader
					}
				}
				add = {
					desc = "minarchism_progress_from_company_productivity_tt"
					every_company = {
						limit = {
							company_is_prosperous = yes
							executive = {
								has_ideology = ideology:ideology_minarchist_leader
							}
						}
						add = {
							add = company_productivity
							divide = 20
						}
					}
				}
			}
			### From Num Prosperous Companies
			if = {
				limit = { num_companies > 0 }
				add = {
					desc = "minarchism_progress_from_average_prosperity_tt"
					add = {
					    add = {
	    					every_company = {
	    						add = {
	    							value = company_prosperity
	    							multiply = company_employed_levels
	    						}
	    					}
	    				}
	    				divide = {
	    					every_company = {
	    						add = company_employed_levels
	    					}
	    				}
	    				multiply = 0.03
					}
					subtract = 0.5
				}
			}
			### Minarchists gain progress from legitimacy in government
			if = {
				limit = {
					any_interest_group = {
						leader = {
							has_ideology = ideology:ideology_minarchist_leader
						}
					}
				}
				add = {
					desc = "minarchism_progress_from_minarchist_legitimacy_tt"
					value = legitimacy
					multiply = 0.05
					subtract = 2.5
				}
			}
			### From Effects (event/button nudges)
			if = {
				limit = {
					exists = var:minarchist_effects_var
					NOT = { var:minarchist_effects_var = 0 }
				}
				add = {
					desc = "minarchism_progress_from_effects_tt"
					value = owner.var:minarchist_effects_var
				}
			}
			if = {
				limit = {
					has_variable = suspended_lobbying_var
				}
				subtract = {
					desc = "progress_suspended_lobbying_tt"
					value = 1
				}
			}
		}
	}
	start_value = 15
	min_value = 0
	max_value = 400
}

				limit = {
					has_law_or_variant = law_type:law_public_schools
				}
				add = {
					value = institution_investment_level:institution_schools
					multiply = 1.33
					multiply = 2
					desc = "INSTITUTION_FUNDING_LEVEL_schools"
				}
			}else_if = {
				limit = {
					has_law_or_variant = law_type:law_religious_schools

				add = {
					value = institution_investment_level:institution_schools
					multiply = 1.18
					multiply = 2
					desc = "INSTITUTION_FUNDING_LEVEL_schools"
				}
			}
		}else_if = {
			limit = {
				has_law_or_variant = law_type:law_public_schools
			}
			add = {
				value = institution_investment_level:institution_schools
				multiply = 2
				desc = "INSTITUTION_FUNDING_LEVEL_schools"
			}
		}else_if = {
			limit = {
				has_law_or_variant = law_type:law_religious_schools

				multiply = 2

				multiply = 1.25
				multiply = 2
				desc = "INSTITUTION_FUNDING_LEVEL_health_system"
			}
		}


		if = {
			limit = {
				has_law_or_variant = law_type:law_old_age_pension
			}
			add = {
				value = institution_investment_level:institution_social_security
				multiply = 1.25
				multiply = 2
				desc = "INSTITUTION_FUNDING_LEVEL_social_security"
			}

			}

				value = -2

			}
		}

		if = {
			limit = {

					has_law_or_variant = law_type:law_private_defense_agencies

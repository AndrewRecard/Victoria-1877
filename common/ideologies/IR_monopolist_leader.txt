	}

		value = 100

			limit = {
				owner.ideology_monopolist_leader_disfavoured_laws > 0
			}
			add = {
				value = owner.ideology_monopolist_leader_disfavoured_laws
				multiply = 8  # 1 less law than mins/mols so slightly more impactful
			}
		}
		if = {
			limit = {
				owner.ideology_monopolist_leader_favoured_laws > 0

			subtract = {
				value = owner.ideology_monopolist_leader_favoured_laws
				multiply = 8
			}
		}

		# more likely if angry (non sov mon)

				ig_approval <= -5

					NOT = {
						has_law = law_type:law_sovereign_monopoly

				}

			}
		}

			limit = {
				ig_approval <= -5

							OR = {
								has_ideology = ideology:ideology_minarchist_leader

							}
						}
						is_powerful = yes
						is_in_government = yes
					}
				}
			}

				value = ig_approval
				multiply = 5
			}
		}
		# less likely if content without sov mon
		if = {
			limit = {
				owner = {
					NOT = {
						has_law = law_type:law_sovereign_monopoly
						has_journal_entry = je_boardroom_schism
					}
				}

			}
			subtract = {
				value = ig_approval
				multiply = 5
			}
		}

			limit = {
				owner = {
					has_law = law_type:law_sovereign_monopoly
				}
				ig_approval >= 5
			}
			add = {
				value = ig_approval
				multiply = 5
			}
		}

			limit = {
					is_interest_group_type = ig_landowners
			}
			add = 50
		}

			limit = {
				is_interest_group_type = ig_petty_bourgeoisie
			}
			add = 25
		}
		# becomes home for PB with fascist movement
		if = {
			limit = {
				is_interest_group_type = ig_petty_bourgeoisie
				owner = {
					any_political_movement = {
						is_political_movement_type = movement_fascist
					}
				}
			}
			add = 25
		}

			limit = {
				is_interest_group_type = ig_intelligentsia
				owner = {
					has_law = law_type:law_scientific_management
				}
			}

		}

			limit = {
				is_interest_group_type = ig_devout
			}
			add = -50
		}

					has_law = law_type:law_sovereign_monopoly
				}

			}

			limit = {
				owner = {
					has_journal_entry = je_boardroom_schism
					has_variable = supporting_monopolism_var
				}
			}
			multiply = 1.25
		}
		# propertarian JE causes an uptick in propertarianism

			limit = {

					OR = {
						has_journal_entry = je_industry_and_integrity
						has_variable = completed_industry_and_integrity
					}
				}
			}

		}
		if = {
			limit = {
				owner = {
					monopolists_failed_schism_trigger = yes
				}
			}
			multiply = 0.5
		}
		if = {
			limit = {
				OR = {
					has_variable = communist_ig_var
					has_variable = chose_vanguardism
					has_variable = chose_anarchism
					has_variable = chose_communism
					has_variable = chose_ethno_nationalism
					has_variable = chose_fascism
				}
			}
			multiply = 0.1
		}
	}

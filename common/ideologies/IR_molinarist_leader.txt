		law_theocracy = strongly_disapprove

	}

		law_right_to_associate = disapprove
		law_factory_councils = strongly_disapprove

		law_peasant_levies = strongly_disapprove
		law_national_militia = neutral
		law_professional_army = disapprove
		law_mass_conscription = strongly_disapprove
		law_private_defense_agencies = approve
	}

		law_closed_borders = disapprove
		law_migration_controls = disapprove
		law_no_migration_controls = neutral
	}

		law_local_police = approve
		law_dedicated_police = disapprove
		law_militarized_police = disapprove
		law_no_police = neutral
	}

	country_trigger = {
		has_law_or_variant = law_type:law_directorate
	}

	interest_group_leader_trigger = {
		ideology_molinarist_valid_trigger = yes
	}

	non_interest_group_leader_trigger = {
		NAND = {
			has_role_of_type = agitator
			owner = {
				NOT = {
					boardroom_schism_resolved_trigger = yes
				}
				has_law = law_type:law_polycentric_law
			}
		}
		interest_group = {
			ideology_molinarist_valid_trigger = yes
		}
	}

	interest_group_leader_weight = {
		value = {
			add = 100
			desc = "base_value"
		}

		if = {
			limit = {
				owner.ideology_molinarist_leader_disfavoured_laws > 0
			}
			add = {
				value = owner.ideology_molinarist_leader_disfavoured_laws
				multiply = 7  # less ideological, more profit driven
				desc = "DISFAVORED_LAWS"
			}
		}
		if = {
			limit = {
				owner.ideology_molinarist_leader_favoured_laws > 0
			}
			subtract = {
				value = owner.ideology_molinarist_leader_favoured_laws
				multiply = 7
				desc = "FAVORED_LAWS"
			}
		}

		# more likely if angry (non polycentric law)
		if = {
			limit = {
				ig_approval <= define:NPolitics|APPROVAL_THRESHOLD_UNHAPPY
				owner = {
					NOT = {
						has_law = law_type:law_polycentric_law
					}
				}
			}
			subtract = {
				value = ig_approval
				multiply = 5
				desc = "IG_APPROVAL_BELOW_UNHAPPY"
			}
		}
		# even more likely if angry under powerful rivals
		if = {
			limit = {
				ig_approval <= define:NPolitics|APPROVAL_THRESHOLD_UNHAPPY
				owner = {
					any_interest_group = {
						leader = {
							OR = {
								has_ideology = ideology:ideology_minarchist_leader
								has_ideology = ideology:ideology_monopolist_leader
							}
						}
						is_powerful = yes
						is_in_government = yes
					}
				}
			}
			subtract = {
				value = ig_approval
				multiply = 5
				desc = "IG_APPROVAL_BELOW_UNHAPPY_ENEMIES"
			}
		}
		# less likely if content without polycentric law
		if = {
			limit = {
				owner = {
					NOT = {
						has_law = law_type:law_polycentric_law
						has_journal_entry = je_boardroom_schism
					}
				}
				ig_approval >= define:NPolitics|APPROVAL_THRESHOLD_HAPPY
			}
			subtract = {
				value = ig_approval
				multiply = 5
				desc = "HAPPY_WITHOUT_POLYCENTRIC_LAW"
			}
		}
		# more likely if content with polycentric law
		if = {
			limit = {
				owner = {
					has_law = law_type:law_polycentric_law
				}
				ig_approval >= define:NPolitics|APPROVAL_THRESHOLD_HAPPY
			}
			add = {
				value = ig_approval
				multiply = 5
				desc = "HAPPY_WITH_POLYCENTRIC_LAW"
			}
		}
		# home of the intelligentsia and rural folk
		if = {
			limit = {
				OR = {
				    is_interest_group_type = ig_intelligentsia
					is_interest_group_type = ig_rural_folk
				}
			}
			add = {
				value = 50
				desc = "INTELLIGENTSIA_OR_RURAL_FOLK"
			}
		}
		# more likely for defense contractors in general
		if = {
			limit = {
				is_interest_group_type = ig_armed_forces
				has_ideology = ideology:ideology_martial_corporatist

			}
			add = {
				value = 25
				desc = "ig_defense_contractors"
			}
		}
		# less likely for PB
		if = {
			limit = {
				is_interest_group_type = ig_petty_bourgeoisie
			}
			add = {
				value = -25
				desc = "ig_petty_bourgeoisie"
			}
		}
		# less likely for PB with fascist movement
		if = {
			limit = {
				is_interest_group_type = ig_petty_bourgeoisie
				owner = {
					any_political_movement = {
						is_political_movement_type = movement_fascist
					}
				}
			}
			add = {
				value = -25
				desc = "PB_FASCIST_MOVEMENT"
			}
		}
		# much less likely for devout
		if = {
			limit = {
				is_interest_group_type = ig_devout
			}
			add = {
				value = -50
				desc = "ig_devout"
			}
		}
		# more likely if in government with polycentric law
		if = {
			limit = {
				owner = {
					has_law = law_type:law_polycentric_law
				}
				is_in_government = yes
			}
			add = {
				value = 50
				desc = "IN_GOVERNMENT_POLYCENTRIC_LAW"
			}
		}
		# slightly more likely if not powerful
		if = {
			limit = {
				is_powerful = no
			}
			add = {
				value = 25
				desc = "IG_NOT_POWERFUL"
			}
		}
		# more likely for defense contractors with no/local police
		if = {
			limit = {
				is_interest_group_type = ig_armed_forces
				has_ideology = ideology:ideology_martial_corporatist
				owner = {
					OR = {
						has_law_or_variant = law_type:law_local_police
						has_law_or_variant = law_type:law_no_police
					}
				}

			}
			multiply = {
				value = 1.5
				desc = "LOCAL_OR_NO_POLICE"
			}
		}
		# more likely if supporting
		if = {
			limit = {
				owner = {
					has_journal_entry = je_boardroom_schism
					has_variable = supporting_molinarism_var
				}
			}
			multiply = {
				value = 1.25
				desc = "SUPPORTING_MOLINARISM"
			}
		}
		# propertarian JE causes an uptick in propertarianism
		if = {
			limit = {
				owner = {
					OR = {
						has_journal_entry = je_industry_and_integrity
						has_variable = completed_industry_and_integrity
					}
				}
			}
			multiply = {
				value = 1.25
				desc = "INDUSTRY_AND_INTEGRITY_EFFECTS"
			}
		}
		if = {
			limit = {
				owner = {
					molinarists_failed_schism_trigger = yes
				}
			}
			multiply = {
				value = 0.5
				desc = "FAILED_SCHISM"
			}
		}
		if = {
			limit = {
				OR = {
					has_variable = communist_ig_var
					has_variable = chose_vanguardism
					has_variable = chose_anarchism
					has_variable = chose_communism
					has_variable = chose_ethno_nationalism
					has_variable = chose_fascism
				}
			}
			multiply = {
				value = 0.1
				desc = "CHOSE_A_DIFFERENT_PATH"
			}
		}
	}

	non_interest_group_leader_weight = {
		value = 100

		if = {
			limit = {
				owner.ideology_molinarist_leader_disfavoured_laws > 0

			}

				multiply = 5
			}
		}
		# even more likely if angry under rivals
		if = {
			limit = {

					any_interest_group = {

							OR = {
								has_ideology = ideology:ideology_minarchist_leader

						is_powerful = yes
						is_in_government = yes
					}
				}
			}
			subtract = {
				value = interest_group.ig_approval
				multiply = 5
			}
		}
		# less likely if content without polycentric law
		if = {
			limit = {
				owner = {
					NOT = {
						has_law = law_type:law_polycentric_law
						has_journal_entry = je_boardroom_schism
					}
				}
				interest_group.ig_approval >= 5
			}
			subtract = {
				value = interest_group.ig_approval
				multiply = 5
			}
		}
		# more likely if content with polycentric law
		if = {
			limit = {
				owner = {
					has_law = law_type:law_polycentric_law
				}
				interest_group.ig_approval >= 5
			}
			add = {
				value = interest_group.ig_approval
				multiply = 5
			}
		}
		# home of the intelligentsia and rural folk
		if = {
			limit = {
				OR = {
				    is_interest_group_type = ig_intelligentsia
					is_interest_group_type = ig_rural_folk
				}
			}
			add = 50

		# more likely for defense contractors in general
		if = {
			limit = {
				is_interest_group_type = ig_armed_forces
				interest_group = {
					has_ideology = ideology:ideology_martial_corporatist
				}
			}
			add = 25
		}
		# less likely for PB
		if = {
			limit = {
				is_interest_group_type = ig_petty_bourgeoisie
			}
			add = -25
		}
		# less likely for PB with fascist movement
		if = {
			limit = {
				is_interest_group_type = ig_petty_bourgeoisie
				owner = {
					any_political_movement = {
						is_political_movement_type = movement_fascist
					}
				}
			}
			add = -25
		}
		# less likely for PB with fascist movement
		if = {
			limit = {
				is_interest_group_type = ig_petty_bourgeoisie
				owner = {
					any_political_movement = {
						is_political_movement_type = movement_fascist
					}
				}
			}
			add = -25
		}
		# much less likely for devout
		if = {
			limit = {
				is_interest_group_type = ig_devout
			}
			add = -50
		}
		# more likely if in government with polycentric law
		if = {
			limit = {
				owner = {
					has_law = law_type:law_polycentric_law
				}
				interest_group = {
					is_in_government = yes
				}
			}
			add = 50
		}
		# slightly more likely if not powerful
		if = {
			limit = {
				interest_group = {
					is_powerful = no
				}
			}
			add = 25
		}
		# more likely for defense contractors with no/local police
		if = {
			limit = {

				}

					OR = {
						has_law_or_variant = law_type:law_local_police
					    has_law_or_variant = law_type:law_no_police
					}
				}
			}
			multiply = 1.5
		}
		# more likely if supporting
		if = {
			limit = {
				owner = {
					has_journal_entry = je_boardroom_schism
					has_variable = supporting_molinarism_var
				}
			}
			multiply = 1.25
		}
		# propertarian JE causes an uptick in propertarianism
		if = {
			limit = {
				owner = {
					OR = {
						has_journal_entry = je_industry_and_integrity
						has_variable = completed_industry_and_integrity
					}
				}
			}
			multiply = 1.25
		}
		if = {
			limit = {
				owner = {
					molinarists_failed_schism_trigger = yes
				}
			}
			multiply = 0.5
		}
		if = {
			limit = {
				interest_group = {
					OR = {
						has_variable = communist_ig_var
						has_variable = chose_vanguardism
						has_variable = chose_anarchism
						has_variable = chose_communism
						has_variable = chose_ethno_nationalism
						has_variable = chose_fascism
					}
				}
			}
			multiply = 0.1

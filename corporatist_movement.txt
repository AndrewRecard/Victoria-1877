movement_propertarian = {
	category = movement_category_ideological

	ideology = ideology_propertarian_movement
	character_ideologies = {
		ideology_propertarian_leader
	}

	creation_trigger = {
		has_technology_researched = corporate_management
		country_should_exclusively_focus_on_modernization = no
	}
	creation_weight = {
		value = 100
		if = {
			limit = {
				owner ?= { # If a movement of this type exists in a neighbouring country, make it more likely to spawn here as well
					any_neighbouring_state = {
						owner = {
							prestige > ROOT.owner.prestige
							any_political_movement = {
								is_political_movement_type = movement_propertarian
							}
						}
					}
				}
			}
			multiply = {
				value = 2
			}
		}
	}
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}

	character_support_trigger = {
		OR = {
			is_interest_group_type = ig_devout
			has_ideology = ideology:ideology_corporatist_leader
			has_ideology = ideology:ideology_fascist
			has_ideology = ideology:ideology_integralist
			has_ideology = ideology:ideology_social_democrat
			has_ideology = ideology:ideology_utopian
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = {
				has_ideology = ideology:ideology_corporatist_leader
			}
			multiply = {
				value = 5
			}
		}
		if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_fascist
					has_ideology = ideology:ideology_integralist
				}
				NOT = {
					owner ?= {
						any_political_movement = {
							is_political_movement_type = movement_fascist
						}
					}
				}
			}
			multiply = {
				value = 3
			}
		}
		if = {
			limit = {
				is_interest_group_type = ig_devout
			}
			multiply = {
				value = 2
			}
		}
	}

	pop_support_trigger = {
		NOT = {
			is_pop_type = slaves
		}
	}

	pop_support_factors = {
		movement_support_high_literacy
		movement_support_aristocrats
		movement_support_clergymen
		movement_support_shopkeepers
		movement_support_clerks
		movement_support_peasants
	}

	pop_support_weight = {
		if = {
			limit = {
				exists = scope:political_movement
				scope:political_movement ?= {
						OR = {
							is_being_suppressed = yes
							is_being_bolstered = yes
						}
					}
				}
			if = {
				limit = {
					scope:political_movement = {
						is_being_suppressed = yes
					}
				}
				add = {
					value = suppression_base_value
					multiply = {
						value = -0.5
						add = owner.modifier:country_suppression_attraction_factor
					}
					desc = "concept_suppressed"
				}
			}
			else = {
				add = {
					value = bolstering_base_value
					multiply = {
						value = 0.5
						add = owner.modifier:country_bolster_attraction_factor
					}
					desc = "concept_bolstered"
				}
			}
		}
		add = {
			value = 1
			desc = "POP_BASE_SUPPORT"
		}

		if = {
			limit = {
				strata = upper
			}
			if = {
				limit = {
					is_pop_type = aristocrats
				}
				add = {
					value = 18
					desc = "POP_ARISTOCRATS"
				}
			}
			else = {
				add = {
					value = 6
					desc = "UPPER_NO_ICON"
				}
			}
			if = {
				limit = {
					owner ?= {
						any_political_movement = {
							OR = {
								is_political_movement_type = movement_socialist
								is_political_movement_type = movement_anarchist
								is_political_movement_type = movement_communist
							}
							political_movement_support >= 0.10 # (chuckles) im in danger
						}
					}
				}
				multiply = {
					value = 1.5
					desc = "STRONG_SOCIALIST_MOVEMENT"
				}
			}
		}
		else_if = {
			limit = {
				strata = middle
			}
			if = {
				limit = {
					is_pop_type = clergymen
				}
				add = {
					value = 25
					desc = "POP_CLERGYMEN"
				}
			}
			else_if = {
				limit = {
					is_pop_type = farmers
				}
				add = {
					value = 15
					desc = "POP_FARMERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = officers
				}
				add = {
					value = 12
					desc = "POP_SHOPKEEPERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = shopkeepers
				}
				add = {
					value = 10
					desc = "POP_SHOPKEEPERS"
				}
			}
			else = {
				add = {
					value = 9
					desc = "MIDDLE_NO_ICON"
				}
			}

			if = {
				limit = {
					owner ?= {
						any_political_movement = {
							OR = {
								is_political_movement_type = movement_socialist
								is_political_movement_type = movement_anarchist
								is_political_movement_type = movement_communist
							}
							political_movement_support >= 0.10
						}
					}
				}
				multiply = {
					value = 1.25
					desc = "STRONG_SOCIALIST_MOVEMENT"
				}
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = soldiers
				}
				add = {
					value = 6
					desc = "POP_CLERKS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = peasants
				}
				add = {
					value = 15
					desc = "POP_PEASANTS"
				}
			}
			else = {
				add = {
					value = 3
					desc = "LOWER_NO_ICON"
				}
			}
		}

		if = {
			limit = {
				standard_of_living < state.average_expected_sol
				strata = middle
			}
			add = { # Increases likelihood of support if impoverished
				desc = "IS_BELOW_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				max = 10
			}
		}
		else_if = {
			limit = {
				standard_of_living > state.average_expected_sol
				strata = middle
			}
			add = { # Decreases likelihood of support if well-off
				desc = "IS_ABOVE_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				min = -5
			}
		}

		if = {
			limit = {
				is_employed = no
			}
			multiply = {
				value = 1.5
				desc = "UNEMPLOYED"
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}

		multiply = { # Intentional that this is different from fascism
			value = 0.01
			multiply = pop_acceptance
			add = 0.20
			desc = "ACCEPTANCE"
		}
	}

	revolution = {
		possible = {
			political_movement_support >= 0.05
		}

		weight = {
			value = 100
		}

		state_weight = {
			value = 1
		}
	}

	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}

		weight = {
			value = 1000
		}
	}

	additional_radicalism_factors = {
		owner = {
			if = {
				limit = { has_institution = institution_schools }

				add = {
					value = 3
					subtract = "institution_investment_level(institution_schools)"
					multiply = 0.05
					desc = "INSTITUTION_FUNDING_LEVEL_schools"
				}
			}
			else = {
				add = {
					value = 0.1
					desc = "NO_INSTITUTION_schools"
				}
			}
		}

		if = {
			limit = {
				owner = {
					any_neighbouring_state = {
						owner = {
							NOT = { THIS = ROOT.owner }
							any_political_movement = {
								is_political_movement_type = movement_corporatist
								is_insurrectionary = yes
							}
						}
					}
				}
			}
			add = {
				desc = "NEIGHBOURING_COUNTRY_WITH_REVOLUTION"
				value = 0.15
			}
		}
	}
}

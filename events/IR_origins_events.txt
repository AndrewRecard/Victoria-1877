	event_image = {
		video = "votp_realist_household"
	}

		has_technology_researched = shareholder_primacy
		NOT = { has_global_variable = molinari_happened }

			name = molinari_happened

			template = agitator_gustave_de_molinari

		}

			place_character_in_void = very_long_modifier_time
		}
	}

		name = propertarian_origins.1.a
		default_option = yes
		if = {
			limit = {
				any_political_movement = {
					NOT = {
						is_political_movement_type = movement_propertarian
					}
				}
			}
			create_political_movement = {
				type = movement_propertarian
			}
		}
		ig:ig_industrialists ?= {
			add_modifier = {
				name = ir_supported_propertarian_modifier
				days = normal_modifier_time
			}
		}
		ig:ig_intelligentsia ?= {
			add_modifier = {
				name = ir_supported_propertarian_modifier
				days = normal_modifier_time
			}
		}

			limit = {
				is_not_in_geographic_region = geographic_region_north_america

				is_agitator = yes

					persistent
				}
			}

			create_character = {
				template = agitator_gustave_de_molinari
			}
		}
	}

	option = {
		name = propertarian_origins.1.b

			limit = {

					NOT = {
						is_political_movement_type = movement_propertarian
					}
				}
			}
			create_political_movement = {
				type = movement_propertarian
			}
		}

			add_modifier = {
				name = ir_promoted_molinari_modifier
				days = normal_modifier_time
			}
		}

			add_modifier = {
				name = ir_promoted_molinari_modifier
				days = normal_modifier_time
			}

			limit = {
				is_not_in_geographic_region = geographic_region_north_america

			}

				}

			}

				name = ir_propertarians_welcomed_modifier

			}

			strata = middle

			scope:origins_propertarian_leader = {
				set_character_busy_and_immortal = no
			}

			base = 1

							has_propertarian_ideology = yes

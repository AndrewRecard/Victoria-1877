defense_incorporation.1 = {
	type = country_event
	placement = root
	title = defense_incorporation.1.t
	desc = defense_incorporation.1.d
	flavor = defense_incorporation.1.f
	duration = 3

	on_opened_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/invention_icons/wargaming.dds"
	event_image = {
		video = "unspecific_signed_contract"
	}

	immediate = {
		random_scope_character = {
			limit = {
				is_ruler_of_own_country = yes
			}
			save_scope_as = head_of_state1
		}
	}

	cancellation_trigger = {
		NOT = {
			has_law_or_variant = law_type:law_private_defense_agencies
		}
	}

	option = {
		name = defense_incorporation.1.a
		default_option = yes
		add_radicals = {
			value = very_large_radicals
			pop_type = officers
		}
		add_modifier = {
			name = ir_defense_agencies_drain_modifier
			years = 5
			is_decaying = yes
		}
	}
}

defense_incorporation.2 = {
	type = country_event
	placement = scope:barracks_state
	title = defense_incorporation.2.t
	desc = defense_incorporation.2.d
	flavor = defense_incorporation.2.f
	duration = 3

	on_opened_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_military.dds"
	event_image = {
		video = "ip2_unruly_soldiers"
	}

	cooldown = {
		days = normal_modifier_time
	}

	trigger = {
		any_scope_state = {
			is_capital = no
			any_scope_building = {
				is_building_type = building_barrack
			}
		}
	}

	immediate = {
		random_scope_state = {
			limit = {
				is_capital = no
				any_scope_building = {
					is_building_type = building_barrack
				}
			}
			weight = {
				factor = 1
				modifier = {
					factor = 2
					trigger = {
					    any_scope_building = {
    						is_building_type = building_barrack
							level >= 10
    					}
					}
				}
				modifier = {
					factor = 2
					trigger = {
						any_scope_building = {
							is_building_type = building_barrack
							level >= 20
						}
					}
				}
				modifier = {
					factor = 2
					trigger = {
						any_scope_building = {
							is_building_type = building_barrack
							level >= 30
						}
					}
				}
				modifier = {
					factor = 2
					trigger = {
						any_scope_building = {
							is_building_type = building_barrack
							level >= 40
						}
					}
				}
			}
			save_scope_as = barracks_state
		}
	}

	option = {
		name = defense_incorporation.2.a
		custom_tooltip = {
			text = defense_incorporation_progress_increase_5_tt
			je:je_defense_incorporation = {
				add_progress = {
					value = 5
					name = ir_defense_incorporation_bar
				}
			}
		}
		scope:barracks_state = {
			add_radicals_in_state = {
				pop_type = officers
				value = large_radicals
			}
		}
	}

	option = {
		name = defense_incorporation.2.b
		default_option = yes
		custom_tooltip = {
			text = defense_incorporation_progress_decrease_5_tt
			je:je_defense_incorporation = {
				add_progress = {
					value = -5
					name = ir_defense_incorporation_bar
				}
			}
		}
		scope:barracks_state = {
			add_loyalists_in_state = {
				pop_type = officers
				value = large_radicals
			}
		}
	}

	cancellation_trigger = {
		NOT = {
			has_journal_entry = je_defense_incorporation
		}
	}
}

	on_opened_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/invention_icons/logistics_tech.dds"
	event_image = {
		video = "ip4_mining_scene"
	}

	cooldown = {
		days = long_modifier_time
	}

	option = {
		name = defense_incorporation.3.a
		if = {
			limit = {
				NOT = {
					has_technology_researched = steel_railway_cars
				}
			}
			add_technology_progress = { technology = steel_railway_cars progress = 6000 }
		}else_if = {
			limit = {
				NOT = {
					has_technology_researched = telephone
				}
			}
			add_technology_progress = { technology = telephone progress = 6000 }
		}else_if = {
			limit = {
				NOT = {
					has_technology_researched = radio
				}
			}
			add_technology_progress = { technology = radio progress = 6000 }
		}else = {
			add_modifier = {
				name = ir_proprietary_tech_modifier
				days = normal_modifier_time
			}
		}
		add_modifier = {
			name = ir_new_logistics_modifier
			days = normal_modifier_time
		}
		ig:ig_armed_forces = {
			add_modifier = {
				name = ir_seized_patents
				days = normal_modifier_time
			}
		}
		custom_tooltip = {
			text = defense_incorporation_progress_decrease_10_tt
			je:je_defense_incorporation = {
				add_progress = {
					value = -10
					name = ir_defense_incorporation_bar
				}
			}
		}
	}

	option = {
		name = defense_incorporation.3.b
		default_option = yes
		add_modifier = {
			name = ir_new_logistics_modifier
			days = normal_modifier_time
		}
	}

	cancellation_trigger = {
		NOT = {
			has_journal_entry = je_defense_incorporation
		}
	}
}

defense_incorporation.4 = {
	type = country_event
	placement = scope:mercenary_state
	title = defense_incorporation.4.t
	desc = defense_incorporation.4.d
	flavor = defense_incorporation.4.f
	duration = 3

	on_opened_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_military.dds"
	event_image = {
		video = "votp_cops_march"
	}

	trigger = {
		any_scope_state = {
			is_incorporated = yes
			has_building = building_barrack
			OR = {
				has_building = building_munition_plant
				has_building = building_arms_industry
				has_building = building_artillery_foundry
			}
		}
	}

	immediate = {
		random_scope_state = {
			limit = {
				is_incorporated = yes
				has_building = building_barrack
				OR = {
					has_building = building_munition_plant
					has_building = building_arms_industry
					has_building = building_artillery_foundry
				}
			}
			weight = {
				factor = 1
				modifier = {
					factor = 3
					trigger = {
						b:building_barrack = {
							level >= 10
						}
						OR = {
							b:building_munition_plant = {
								level >= 5
							}
							b:building_arms_industry = {
								level >= 5
							}
							b:building_artillery_foundry = {
								level >= 5
							}
						}
					}
				}
				modifier = {
					factor = 3
					trigger = {
						b:building_barrack = {
							level >= 20
						}
						OR = {
							b:building_munition_plant = {
								level >= 10
							}
							b:building_arms_industry = {
								level >= 10
							}
							b:building_artillery_foundry = {
								level >= 10
							}
						}
					}
				}
			}
			save_scope_as = mercenary_state
		}
	}

	cooldown = {
		days = normal_modifier_time
	}

	option = {
		name = defense_incorporation.4.a
		add_modifier = {
			name = ir_local_authority_modifier
			days = normal_modifier_time
		}
		scope:mercenary_state = {
			add_modifier = {
				name = ir_state_crackdown_modifier
				days = normal_modifier_time
			}
			add_radicals_in_state = {
				pop_type = officers
				value = medium_radicals
			}
			add_radicals_in_state = {
				pop_type = soldiers
				value = large_radicals
			}
			add_devastation = 5
		}
		custom_tooltip = {
			text = defense_incorporation_progress_decrease_5_tt
			je:je_defense_incorporation = {
				add_progress = {
					value = -5
					name = ir_defense_incorporation_bar
				}
			}
		}
	}

	option = {
		name = defense_incorporation.4.b
		default_option = yes
		scope:mercenary_state = {
			add_loyalists_in_state = {
				pop_type = soldiers
				value = large_radicals
			}
			add_radicals_in_state = {
				strata = lower
				value = medium_radicals
			}
			add_devastation = 5
		}
		add_modifier = {
			name = ir_unrestrained_defense_agencies_modifier
			days = normal_modifier_time
		}
		custom_tooltip = {
			text = defense_incorporation_progress_increase_5_tt
			je:je_defense_incorporation = {
				add_progress = {
					value = 5
					name = ir_defense_incorporation_bar
				}
			}
		}
	}

	cancellation_trigger = {
		NOT = {
			has_journal_entry = je_defense_incorporation
		}
	}
}

defense_incorporation.6 = {
	type = country_event
	placement = root
	title = defense_incorporation.6.t
	desc = defense_incorporation.6.d
	flavor = defense_incorporation.6.f
	duration = 3

	on_opened_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/invention_icons/wargaming.dds"
	event_image = {
		video = "ep1_transfer_of_authority"
	}

	immediate = {
	}

	cancellation_trigger = {
		NOT = {
			has_law_or_variant = law_type:law_private_defense_agencies
		}
	}

	option = {
		name = defense_incorporation.6.a
		default_option = yes
		ir_defense_contractors_evolution_effect = yes
		add_modifier = {
			name = ir_defense_agencies_investment_modifier
			years = 10
		}
		remove_modifier = ir_defense_agencies_drain_modifier
		every_scope_character = {
			limit = {
				has_role_of_type = general
				propertarian_or_market_liberal = no
				loyalty >= 75
			}
			if = {
				limit = {
					owner = {
						NOT = {
						    any_political_movement = {
								is_political_movement_type = movement_propertarian
							}
						}
					}
				}
				set_ideology = ideology:ideology_market_liberal
			}else_if = {
				limit = {
					owner = {
						NOT = {
							has_law_or_variant = law_type:law_directorate
						}
					}
				}
				random_list = {
					50 = {
						set_ideology = ideology:ideology_market_liberal
					}
					50 = {
						set_ideology = ideology:ideology_minarchist_leader
					}
				}
			}else_if = {
				limit = {
					owner = {
						has_law = law_type:law_dedicated_police
					}
				}
				random_list = {
					50 = {
						set_ideology = ideology:ideology_minarchist_leader
					}
					25 = {
						set_ideology = ideology:ideology_molinarist_leader
					}
					25 = {
						set_ideology = ideology:ideology_monopolist_leader
					}
				}
			}else_if = {
				limit = {
					owner = {
						has_law = law_type:law_militarized_police
					}
				}
				random_list = {
					50 = {
						set_ideology = ideology:ideology_monopolist_leader
					}
					25 = {
						set_ideology = ideology:ideology_molinarist_leader
					}
					25 = {
						set_ideology = ideology:ideology_minarchist_leader
					}
				}
			}else = {
				random_list = {
					50 = {
						set_ideology = ideology:ideology_molinarist_leader
					}
					25 = {
						set_ideology = ideology:ideology_monopolist_leader
					}
					25 = {
						set_ideology = ideology:ideology_minarchist_leader
					}
				}
			}
		}
		every_scope_character = {
			limit = {
				has_role_of_type = general
				propertarian_or_market_liberal = no
				loyalty < 75
			}
			remove_character_role = general
		}
	}

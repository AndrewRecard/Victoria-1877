namespace = boardroom_schism

# Pick an ideology, spawns a weighted-ig agitator if not present
boardroom_schism.1 = {
	type = country_event
	title = boardroom_schism.1.t
	desc = boardroom_schism.1.d
	flavor = boardroom_schism.1.f
	duration = 3

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	cooldown = { days = normal_modifier_time }

	immediate = {
		ordered_scope_character = {
			limit = {
				has_propertarian_ideology = yes
			}
			order_by = prominence
			position = 0
			save_scope_as = prominent_propertarian
		}
	}

	gui_window = event_window_1char_tabloid
	left_icon = scope:prominent_propertarian

	option = {
		name = boardroom_schism.1.a
		set_variable = chose_initial_propertarian_philosophy
		custom_tooltip = {
			text = supporting_molinarism_tt
		}
		spawn_molinarist_agitator_effect = yes
		custom_tooltip = {
			text = molinarism_progress_increase_10_tt
			je:je_boardroom_schism = {
				add_progress = { value = 10 name = schism_molinarism_bar }
			}
		}
		set_variable = supporting_molinarism_var
	}

	option = {
		name = boardroom_schism.1.b
		set_variable = chose_initial_propertarian_philosophy
		custom_tooltip = {
			text = supporting_minarchism_tt
		}
		spawn_monopolist_agitator_effect = yes
		custom_tooltip = {
			text = monopolism_progress_increase_10_tt
			je:je_boardroom_schism = {
				add_progress = { value = 10 name = schism_monopolism_bar }
			}
		}
		set_variable = supporting_monopolism_var
	}

	option = {
		name = boardroom_schism.1.c
		default_option = yes
		set_variable = chose_initial_propertarian_philosophy
		custom_tooltip = {
			text = supporting_minarchism_tt
		}
		spawn_minarchist_agitator_effect = yes
		custom_tooltip = {
			text = minarchism_progress_increase_10_tt
			je:je_boardroom_schism = {
				add_progress = { value = 10 name = schism_minarchism_bar }
			}
		}
		set_variable = supporting_minarchism_var
	}

	cancellation_trigger = {
		NOT = {
			has_journal_entry = je_boardroom_schism
		}
	}
}

	duration = 3

		}
		set_variable = {
			name = random_minarchism_progress
			value = {
				integer_range = {
					min = -50
					max = 75
				}
			}
		}
		set_variable = {
			name = random_molinarism_progress
			value = {
				integer_range = {
					min = -50
					max = 75
				}
			}
		}
	}

	option = {
		name = boardroom_schism.3.a
		default_option = yes
		add_modifier = {
			name = ir_boardroom_confidence_modifier
			days = normal_modifier_time
		}

		custom_tooltip = {
			text = monopolism_progress_custom_tt
			je:je_boardroom_schism = {
				add_progress = { value = root.var:random_monopolism_progress name = schism_monopolism_bar }
			}
		}
		custom_tooltip = {
			text = minarchism_progress_custom_tt
			je:je_boardroom_schism = {
				add_progress = { value = root.var:random_minarchism_progress name = schism_minarchism_bar }
			}
		}
		custom_tooltip = {
			text = molinarism_progress_custom_tt
			je:je_boardroom_schism = {
				add_progress = { value = root.var:random_molinarism_progress name = schism_molinarism_bar }
			}
		}
	}

	option = {
		name = boardroom_schism.3.b
		add_modifier = {
			name = ir_offbook_liabilities_modifier
			days = short_modifier_time
		}
		add_radicals = {
			value = small_radicals
		}
		remove_modifier = ir_sponsored_thinktanks_modifier
	}

	cancellation_trigger = {
		OR = {
			has_variable = boardroom_schism_freeze
			NOT = {
				has_journal_entry = je_boardroom_schism
			}
		}
	}

	after = {
		remove_variable = random_minarchism_progress
		remove_variable = random_molinarism_progress
		remove_variable = random_monopolism_progress
	}
}

boardroom_schism.4 = {
	type = country_event

	desc = boardroom_schism.4.d
	flavor = boardroom_schism.4.f
	duration = 3

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_scales.dds"
	event_image = { video = "europenorthamerica_capitalists_meeting" }

	immediate = {
		ordered_interest_group = {
			limit = {
				leader = {
					has_ideology = ideology:ideology_monopolist_leader
				}
			}
			order_by = ig_clout
			position = 0
			check_range_bounds = no
			leader = {
				save_scope_as = mandate_leader
			}
		}
	}

	option = {
		name = boardroom_schism.4.a
		default_option = yes
		every_interest_group = {
			limit = {
				ig_counts_as_marginal = no
				leader = {
					has_ideology = ideology:ideology_monopolist_leader
				}
			}
			add_modifier = {
				name = ir_ig_mandate_modifier
				days = short_modifier_time
			}
		}
		add_modifier = {
			name = ir_schism_mandate_modifier
			days = short_modifier_time
		}
		random_political_movement = {
			limit = {
				is_political_movement_type = movement_propertarian
			}
			set_core_ideology = ideology_propertarian_monopolist
		}
	}

	option = {
		name = boardroom_schism.4.b
		add_modifier = {
			name = ir_mandate_opposed_authority_modifier
			days = short_modifier_time
		}
		every_interest_group = {
			limit = {
				ig_counts_as_marginal = no
				leader = {
					OR = {
						has_ideology = ideology:ideology_minarchist_leader
						has_ideology = ideology:ideology_molinarist_leader
					}
				}
			}
			add_modifier = {
				name = ir_mandate_opposed_ig_modifier
				days = short_modifier_time
			}
		}
		add_radicals = {
			value = large_radicals
		}
		random_political_movement = {
			limit = {
				is_political_movement_type = movement_propertarian
			}
			set_core_ideology = ideology_propertarian_monopolist
		}
	}

	cancellation_trigger = {
		NOT = {
			has_journal_entry = je_monopolist_mandate
		}
	}
}

boardroom_schism.7 = {
	type = country_event
	title = boardroom_schism.7.t
	desc = boardroom_schism.7.d
	flavor = boardroom_schism.7.f
	duration = 3

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	cooldown = { days = normal_modifier_time }

	gui_window = event_window_1char_tabloid
	left_icon = scope:defector_executive

	trigger = {
		NOR = {
			has_variable = boardroom_schism_freeze
			has_variable = paused_lobbying_var
		}
		any_company = {
			company_is_prosperous = yes
			executive = {
				has_propertarian_ideology = yes
			}
		}
		calc_true_if = {
			amount >= 2
			je:je_boardroom_schism = {
				scripted_bar_progress = { name = schism_monopolism_bar value >= 40 }
			}
			je:je_boardroom_schism = {
				scripted_bar_progress = { name = schism_minarchism_bar value >= 40 }
			}
			je:je_boardroom_schism = {
				scripted_bar_progress = { name = schism_molinarism_bar value >= 40 }
			}
		}
	}

	immediate = {
		random_company = {
			limit = {
				company_is_prosperous = yes
				executive = {
					has_propertarian_ideology = yes
				}
			}
			save_scope_as = defector_company
			executive = {
				save_scope_as = defector_executive
				interest_group = {
					save_scope_as = defector_ig
				}
			}
		}
		if = {
			limit = { scope:defector_executive = { has_ideology = ideology:ideology_monopolist_leader } }
			set_variable = defector_from_monopolist_var
			random_list = {
				50 = { set_variable = defector_to_minarchist_var }
				50 = { set_variable = defector_to_molinarist_var }
			}
		}
		if = {
			limit = { scope:defector_executive = { has_ideology = ideology:ideology_minarchist_leader } }
			set_variable = defector_from_minarchist_var
			random_list = {
				50 = { set_variable = defector_to_monopolist_var }
				50 = { set_variable = defector_to_molinarist_var }
			}
		}
		if = {
			limit = { scope:defector_executive = { has_ideology = ideology:ideology_molinarist_leader } }
			set_variable = defector_from_molinarist_var
			random_list = {
				50 = { set_variable = defector_to_monopolist_var }
				50 = { set_variable = defector_to_minarchist_var }
			}
		}
	}

	option = {
		name = boardroom_schism.7.a
		default_option = yes
		if = {
			limit = { has_variable = defector_from_monopolist_var }
			custom_tooltip = {
				text = monopolism_progress_decrease_25_tt
				je:je_boardroom_schism = {
					add_progress = { value = -25 name = schism_monopolism_bar }
				}
			}
		}
		if = {
			limit = { has_variable = defector_from_minarchist_var }
			custom_tooltip = {
				text = minarchism_progress_decrease_25_tt
				je:je_boardroom_schism = {
					add_progress = { value = -25 name = schism_minarchism_bar }
				}
			}
		}
		if = {
			limit = { has_variable = defector_from_molinarist_var }
			custom_tooltip = {
				text = molinarism_progress_decrease_25_tt
				je:je_boardroom_schism = {
					add_progress = { value = -25 name = schism_molinarism_bar }
				}
			}
		}
		if = {
			limit = { has_variable = defector_to_monopolist_var }
			custom_tooltip = {
				text = monopolism_progress_increase_25_tt
				je:je_boardroom_schism = {
					add_progress = { value = 25 name = schism_monopolism_bar }
				}
			}
			scope:defector_executive = {
				set_ideology = ideology:ideology_monopolist_leader
			}
		}
		if = {
			limit = { has_variable = defector_to_minarchist_var }
			custom_tooltip = {
				text = minarchism_progress_increase_25_tt
				je:je_boardroom_schism = {
					add_progress = { value = 25 name = schism_minarchism_bar }
				}
			}
			scope:defector_executive = {
				set_ideology = ideology:ideology_minarchist_leader
			}
		}
		if = {
			limit = { has_variable = defector_to_molinarist_var }
			custom_tooltip = {
				text = molinarism_progress_increase_25_tt
				je:je_boardroom_schism = {
					add_progress = { value = 25 name = schism_molinarism_bar }
				}
			}
			scope:defector_executive = {
				set_ideology = ideology:ideology_molinarist_leader
			}
		}
		add_modifier = {
			name = ir_ideological_flagship_modifier
			days = normal_modifier_time
		}
	}

	option = {
		name = boardroom_schism.7.b
		if = {
			limit = { has_variable = defector_from_monopolist_var }
			every_interest_group = {
				limit = {
					leader = { has_ideology = ideology:ideology_monopolist_leader }
				}
				add_modifier = {
					name = ir_discredited_apostate_modifier
					days = normal_modifier_time
				}
			}
		}
		if = {
			limit = { has_variable = defector_from_minarchist_var }
			every_interest_group = {
				limit = {
					leader = { has_ideology = ideology:ideology_minarchist_leader }
				}
				add_modifier = {
					name = ir_discredited_apostate_modifier
					days = normal_modifier_time
				}
			}

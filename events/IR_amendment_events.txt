namespace = amendment_events
amendment_events.1 = {
    type = country_event
    placement = root
    title = amendment_events.1.t
    desc = amendment_events.1.d
    flavor = amendment_events.1.f
    duration = 3

    on_opened_soundeffect = "event:/SFX/UI/Alerts/event_appear"
    icon = "gfx/interface/icons/goods_icons/unused/food.dds"
    event_image = {
        video = "europenorthamerica_rich_and_poor"
    }

    cancellation_trigger = {
        NOR = {
            has_law_or_variant = law_type:law_wage_subsidies
            has_law_or_variant = law_type:law_old_age_pension
        }
    }
    # Ok, we have it now
    option = {
        name = amendment_events.1.a
        default_option = yes
        add_loyalists = {
            value = medium_radicals
            interest_group = ig:ig_trade_unions
        }
        apply_entrenched_entitlements_effect = yes
    }
    # We should get old age pension
    option = {
        name = amendment_events.1.b
        trigger = {
            has_law = law_type:law_wage_subsidies
            has_technology_researched = human_rights
            any_interest_group = {
                is_interest_group_type = ig_trade_unions
                is_in_government = yes
            }
        }
        apply_entrenched_entitlements_effect = yes

        random_law = {
            limit = {
                type = law_type:law_old_age_pension
            }
            save_temporary_scope_as = promised_law
        }
        set_variable = {
            name = desired_law_var
            value = scope:promised_law
        }
        ig:ig_trade_unions = {
            set_variable = petitioning_ig
        }
        add_journal_entry = {
            type = je_government_petition
        }
        add_modifier = {
            name = ir_bolstered_trade_unions_modifier
            years = 5
        }
    }
    # this is getting out of hand
    option = {
        name = amendment_events.1.c
        change_institution_investment_level = {
            institution = institution_social_security
            investment = -1
        }
        add_modifier = {
            name = ir_repeal_welfare_modifier
            years = 2
        }
        add_radicals = {
            value = medium_radicals
            interest_group = ig:ig_trade_unions
        }
    }
}
